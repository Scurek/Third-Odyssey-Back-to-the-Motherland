

##Adopt Iroquois Culture
country_decisions = {

	iroquois_culture = {
		major = yes
		potential = {
			NOT = { exists = IRO }
			culture_group = iroquoian
			NOT = { primary_culture = iroquois }
			OR = {
				any_owned_province = {
					AND = {
						is_core = ROOT
						culture = iroquois
					}
				}
				any_neighbor_country = {
					AND = {
						primary_culture = iroquois
					}
				}
			}
			OR = {
				ai = no
				tag = XXE
				tag = XXF
			}
		}
		allow = {
			is_at_war = no
		}
		effect = {
			change_primary_culture = iroquois
			
			every_owned_province = {
				limit = {
					AND = {
						is_core = ROOT
						NOT = { culture = iroquois }
						culture_group = iroquoian
					}
				}
				change_culture = iroquois
			}
			every_owned_province = {
				limit = {
					AND = {
						culture = iroquois
						NOT = { is_core = ROOT }
					}
				}
				add_core = ROOT
			}
			add_country_modifier = {
				name = "easterntribes_cultural_union"
				duration = 3650
			}
		}
		ai_will_do = { factor = 1 }
	}
