nhs_15_local_aggression = {
		OR = {
			AND = {
				capital_scope = {
					OR = {
						region = great_lakes_region
						region = northeast_america_region
					}
				}
				REB = { check_variable = { which = nhs_iro_aggression value = 15 } }
			}
			AND = {
				capital_scope = {
					OR = {
						region = great_plains_region
						region = mississippi_region
						region = southeast_america_region
					}
				}
				REB = { check_variable = { which = nhs_cah_aggression value = 15 } }
			}
			AND = {
				capital_scope = {
					OR = {
						region = canada_region
						area = james_bay_area
						area = hudson_bay_area
						region = southeast_america_region
					}
				}
				REB = { check_variable = { which = nhs_north_aggression value = 15 } }
			}
			AND = {
				capital_scope = {
					OR = {
						AND = {
							region = hudson_bay_region
							NOT = { area = james_bay_area }
							NOT = { area = hudson_bay_area }
						}
						region = cascadia_region
					}
				}
				REB = { check_variable = { which = nhs_west_aggression value = 15 } }
			}
			AND = {
				capital_scope = {
					OR = {
						region = california_region
						region = mexico_region
					}
				}
				REB = { check_variable = { which = nhs_south_aggression value = 15 } }
			}
		}
}
