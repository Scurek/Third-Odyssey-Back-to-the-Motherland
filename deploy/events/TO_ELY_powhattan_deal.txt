namespace = nhs_powhatan_deal

#Elysians offer deal
country_event = {
	id = nhs_powhatan_deal.2
	title = nhs_powhatan_deal.2.t
	desc = nhs_powhatan_deal.2.d
	picture = ELYEXPLORER_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			capital_scope = {
				save_event_target_as = nhs_pow_mig_origin
			}
			if = {
				limit = {
					932 = {
						is_empty = yes
					}
				}
				932 = {
					save_event_target_as = nhs_pow_mig_tar
				}
			}
			else_if = {
				limit = {
					932 = {
						any_empty_neighbor_province = {
							NOT = { nhs_homelands = yes }
							is_empty = yes
						}
					}
				}
				932 = {
					random_empty_neighbor_province = {
						save_event_target_as = nhs_pow_mig_tar
					}
				}
			}
			else_if = {
				limit = {
					POW = {
						any_owned_province = {
							any_empty_neighbor_province = {
								NOT = { nhs_homelands = yes }
							}
						}
					}
				}
				POW = {
					random_owned_province = {
						limit = {
							any_empty_neighbor_province = {
								NOT = { nhs_homelands = yes }
							}
						}
						random_empty_neighbor_province = {
							save_event_target_as = nhs_pow_mig_tar
						}
					}
				}
			}
			else_if = {
				limit = {
					NHS = {
						any_owned_province = {
							any_empty_neighbor_province = {
								NOT = { nhs_homelands = yes }
							}
						}
					}
				}
				NHS = {
					random_owned_province = {
						limit = {
							any_empty_neighbor_province = {
								NOT = { nhs_homelands = yes }
							}
						}
						random_empty_neighbor_province = {
							save_event_target_as = nhs_pow_mig_tar
						}
					}
				}
			}
		}
	}
	
	option = {
		name = nhs_powhatan_deal.2.a # Hell yea!
		trigger = {
			NOT = { num_of_cities = 2 }
		}
		ai_chance = { factor = 100 }
		event_target:nhs_pow_mig_tar = {
			cede_province = POW
			add_core = ROOT
			change_culture = ROOT
			change_religion = ROOT
			ROOT = {
				set_capital = PREV
			}
			add_institution_embracement = {
				which = "feudalism"
				value = 25
			}
			hidden_effect = {
				to_nat_move_local_tribal_development = {
					ORIGIN = event_target:nhs_pow_mig_origin
					DESTINATION = event_target:nhs_pow_mig_tar
				}
			}
		}
		add_treasury = 200
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		tooltip = {
			if = {
				limit = {
					NOT = { mil_tech = 2 }
				}
				add_mil_tech = 2
			}
			else_if = {
				limit = {
					NOT = { mil_tech = 3 }
				}
				add_mil_tech = 1
			}
			if = {
				limit = {
					is_subject = yes
				}
				grant_independence = yes
				create_alliance = NHS
			}
		}
		to_nat_start_legalism_spread = yes
		every_province = {
			limit = {
				owned_by = NHS
				is_claim = ROOT
			}
			remove_claim = ROOT
		}
		NHS = { country_event = { id = nhs_powhatan_deal.3 days = 1 } }
	}
	option = {
		name = nhs_powhatan_deal.2.b # No
		ai_chance = { factor = 0 }
		NHS = { country_event = { id = nhs_powhatan_deal.4 days = 1 } }
	}
}

#Powhatan accepts deal
country_event = {
	id = nhs_powhatan_deal.3
	title = nhs_powhatan_deal.3.t
	desc = nhs_powhatan_deal.3.d
	picture = NATIVE_DEBATE_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			add_core = 952
			952 = {
				change_culture = ROOT
				change_religion = ROOT
				cede_province = NHS
				remove_core = POW
			}
			set_country_flag = nhs_pow_deal_done
		}
	}
	
	option = {
		name = EXCELLENT
		tooltip = {
			add_core = 952
			952 = {
				change_culture = ROOT
				change_religion = ROOT
				cede_province = NHS
				remove_core = POW
			}
		}
		952 = {
			# add_permanent_province_modifier = {
			# 	name = to_nat_province_depopulated
			# 	duration = 3650
			# }
			add_province_modifier = {
				name = nhs2_settling
				duration = 1825
			}
		}
		FROM = {
			if = {
				limit = {
					NOT = { mil_tech = 2 }
				}
				add_mil_tech = 2
			}
			else_if = {
				limit = {
					NOT = { mil_tech = 3 }
				}
				add_mil_tech = 1
			}
			if = {
				limit = {
					is_subject = yes
				}
				grant_independence = yes
				create_alliance = ROOT
				hidden_effect = {
					every_province = {
						limit = {
							owned_by = POW
						}
						discover_country = ROOT
					}
				}
			}
			to_nat_start_legalism_spread = yes
		}
	}
}

#Initial Event
country_event = {
	id = nhs_powhatan_deal.5
	title = nhs_powhatan_deal.5.t
	desc = nhs_powhatan_deal.5.d
	picture = NATIVE_DEBATE_eventPicture
	
	fire_only_once = yes
		
	trigger = {
		nhs_condition_all_ely = { KOMANDA = tag }
		exists = POW
		owns = 953
		has_discovered = 952
		POW = {
			is_neighbor_of = ROOT
			owns_core_province = 952
			OR = {
				is_subject = no
				is_subject_of = ROOT
			}
		}
		NOT = { has_country_flag = nhs_pow_deal_done }
	}

	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = "nhs_powhatan_deal.5.a"
		custom_tooltip = nhs_unlock_powhatan_deal_dec_tt
		set_country_flag = nhs_enable_powhatan_deal
	}
}
