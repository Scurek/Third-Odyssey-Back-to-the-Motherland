#Trade weapons/money for land with Powhatan
country_event = {
	id = nhs_powhatan_deal.1
	title = nhs_powhatan_deal.1.t
	desc = nhs_powhatan_deal.1.d
	picture = ELYEXPLORER_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		always = no #Now a decision
		tag = NHS
		num_of_cities = 2
		has_discovered = 952
		is_at_war = no
		POW = {
			is_at_war = no
			owns_core_province = 952
			NOT = { num_of_cities = 2 }
		}
	}
	
	option = {
		name = nhs_powhatan_deal.1.a # Seal the deal
		trigger = {
			treasury = 175
		}
		add_treasury = -175
		POW = { country_event = { id = nhs_powhatan_deal.2 days = 1 } }
	}
	
	option = {
		name = nhs_powhatan_deal.1.b # We can't afford it
		ai_chance = {
			factor = 0
		}
	}
}

#Elysians offer deal
country_event = {
	id = nhs_powhatan_deal.2
	title = nhs_powhatan_deal.2.t
	desc = nhs_powhatan_deal.2.d
	picture = ELYEXPLORER_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			capital_scope = {
				save_event_target_as = nhs_pow_mig_origin
			}
			if = {
				limit = {
					932 = {
						is_empty = yes
					}
				}
				932 = {
					save_event_target_as = nhs_pow_mig_tar
				}
			}
			else_if = {
				limit = {
					932 = {
						any_empty_neighbor_province = {
							NOT = { nhs_homelands = yes }
							is_empty = yes
						}
					}
				}
				932 = {
					random_empty_neighbor_province = {
						save_event_target_as = nhs_pow_mig_tar
					}
				}
			}
			else_if = {
				limit = {
					POW = {
						any_owned_province = {
							any_empty_neighbor_province = {
								NOT = { nhs_homelands = yes }
							}
						}
					}
				}
				POW = {
					random_owned_province = {
						limit = {
							any_empty_neighbor_province = {
								NOT = { nhs_homelands = yes }
							}
						}
						random_empty_neighbor_province = {
							save_event_target_as = nhs_pow_mig_tar
						}
					}
				}
			}
			else_if = {
				limit = {
					NHS = {
						any_owned_province = {
							any_empty_neighbor_province = {
								NOT = { nhs_homelands = yes }
							}
						}
					}
				}
				NHS = {
					random_owned_province = {
						limit = {
							any_empty_neighbor_province = {
								NOT = { nhs_homelands = yes }
							}
						}
						random_empty_neighbor_province = {
							save_event_target_as = nhs_pow_mig_tar
						}
					}
				}
			}
		}
	}
	
	option = {
		name = nhs_powhatan_deal.2.a # Hell yea!
		trigger = {
			NOT = { num_of_cities = 2 }
		}
		ai_chance = { factor = 100 }
		event_target:nhs_pow_mig_tar = {
			cede_province = POW
			add_core = ROOT
			change_culture = ROOT
			change_religion = ROOT
			ROOT = {
				set_capital = PREV
			}
			add_institution_embracement = {
				which = "feudalism"
				value = 25
			}
			hidden_effect = {
				to_nat_move_local_tribal_development = {
					ORIGIN = event_target:nhs_pow_mig_origin
					DESTINATION = event_target:nhs_pow_mig_tar
				}
			}
		}
		add_treasury = 200
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		tooltip = {
			if = {
				limit = {
					NOT = { mil_tech = 2 }
				}
				add_mil_tech = 2
			}
			else_if = {
				limit = {
					NOT = { mil_tech = 3 }
				}
				add_mil_tech = 1
			}
			if = {
				limit = {
					is_subject = yes
				}
				grant_independence = yes
				create_alliance = NHS
			}
		}
		to_nat_start_legalism_spread = yes
		NHS = { country_event = { id = nhs_powhatan_deal.3 days = 1 } }
	}
	option = {
		name = nhs_powhatan_deal.2.b # No
		ai_chance = { factor = 0 }
		NHS = { country_event = { id = nhs_powhatan_deal.4 days = 1 } }
	}
}

#Powhatan accepts deal

	id = nhs_powhatan_deal.3
	title = nhs_powhatan_deal.3.t
	desc = nhs_powhatan_deal.3.d
	picture = NATIVE_DEBATE_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			add_core = 952
			952 = {
				change_culture = ROOT
				change_religion = ROOT
				cede_province = NHS
				remove_core = POW
			}
		}
	}
	
	option = {
		name = EXCELLENT
		tooltip = {
			add_core = 952
			952 = {
				change_culture = ROOT
				change_religion = ROOT
				cede_province = NHS
				remove_core = POW
			}
		}
		# 952 = {
		# 	add_province_modifier = {
		# 		name = nhs2_settling
		# 		duration = 1095
		# 	}
		# }
		FROM = {
			if = {
				limit = {
					NOT = { mil_tech = 2 }
				}
				add_mil_tech = 2
			}
			else_if = {
				limit = {
					NOT = { mil_tech = 3 }
				}
				add_mil_tech = 1
			}
			if = {
				limit = {
					is_subject = yes
				}
				grant_independence = yes

					every_province = {
						limit = {
							owned_by = POW
						}
						discover_country = ROOT
					}
				}
			}
			to_nat_start_legalism_spread = yes
		}
	}
}

	title = nhs_powhatan_deal.5.t
	desc = nhs_powhatan_deal.5.d

	fire_only_once = yes

		ai = no

			}
		}
		NOT = { has_global_flag = nhs_pow_dead_done }
	}
	
	option = {
		name = "nhs_powhatan_deal.5.a"

		set_country_flag = nhs_enable_powhatan_deal
	}
