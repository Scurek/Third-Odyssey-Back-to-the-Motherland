	title = nhs_othersevents.51.t
	desc = nhs_othersevents.51.d
	picture = ELYSHIP_eventPicture
	
	is_triggered_only = yes

    trigger = {
		tag = CM1
	}

	immediate = {
		hidden_effect = {
			481 = {
				change_trade_goods = fish
				change_culture = ROOT
				change_religion = ROOT
				add_base_tax = 1
				change_province_name = "Amissae Mare" 
				rename_capital = "Lanthanopolis"
			}
			light_ship = 481
			1572 = { change_province_name = "Sea of the Lost" }
			discover_province = 1572
			add_country_modifier = {
				name = "nhs_handicap_settlers"
				duration = 7300
			}
		}
	}

	option = {	
		name = "NHSOTHERSEVENTSNAME51"
		define_advisor = {
			type = treasurer
			name = "Anna Notaras"
			skill = 1
			discount = yes
			female = yes
			religion = orthodox
			culture = greek
		}
		add_treasury = 50
	}
}

# Event that launches CM1 colonies
province_event = {
	id = nhs_othersevents.54
	title = nhs_othersevents.54.t
	desc = {
		trigger = {
			NOT = { has_global_flag = nhs_ctm_allowed_to_exist }
		}
		desc = nhs_othersevents.54.da
	}
	desc = {
		trigger = {
			has_global_flag = nhs_ctm_allowed_to_exist
		}
		desc = nhs_othersevents.54.db
	}
	picture = ELYSHIP_eventPicture

	fire_only_once = yes
	
	trigger = {
		owner = {
			tag = CM1
			num_of_cities = 1
			OR = {
				has_country_flag = nhs_hadevent_othersevents53
				has_global_flag = nhs_ctm_allowed_to_exist
			}
			# Double check
			NOT = { has_country_flag = nhs_hadevent_othersevents54 }
		}
		is_capital = yes
	}

	mean_time_to_happen = {
		months = 2
	}
	
	immediate = {
		owner = { set_country_flag = nhs_hadevent_othersevents54 }
		hidden_effect = {
			bahama_channel_area = {
				every_country = {
					limit = {
						OR = {
							tag = ROOT
							overlord_of = ROOT
						}
					}
					discover_province = PREV
				}
			}
			caribbean_sea_area = {
				every_country = {
					limit = {
						OR = {
							tag = ROOT
							overlord_of = ROOT
						}
					}
					discover_province = PREV
				}
			}
			discover_province = 1504
			discover_province = 1571
			discover_province = 1572
			discover_province = 1574
			discover_province = 1526
			discover_province = 1575
			if = {
				limit = {
					is_subject = yes
				}
				overlord = {
					discover_province = 1504
					discover_province = 1571
					discover_province = 1572
					discover_province = 1574
					discover_province = 1526
					discover_province = 1575
				}
			}
			carribeans_region = {
				every_country = {
					limit = {
						OR = {
							tag = ROOT
							overlord_of = ROOT
						}
					}
					discover_province = PREV
				}
			}
			if = {
				limit = {
					any_province = {
						OR = {
							area = cuba_area
							area = east_cuba_area
							area = dominica_area
							area = hispaniola_area
						}
						is_empty = yes
						any_empty_neighbor_province = {
							is_empty = yes
						}
					}
				}
				random_province = {
					limit = {
						OR = {
							area = cuba_area
							area = east_cuba_area
							area = dominica_area
							area = hispaniola_area
						}
						is_empty = yes
						any_empty_neighbor_province = {
							is_empty = yes
						}
					}
					create_colony = 1000
					add_core = CM1
				}
			}
			else_if = {
				limit = {
					any_province = {
						region = carribeans_region
						is_empty = yes
						any_empty_neighbor_province = {
							is_empty = yes
						}
					}
				}
				random_province = {
					limit = {
						region = carribeans_region
						is_empty = yes
						any_empty_neighbor_province = {
							is_empty = yes
						}
					}
					create_colony = 1000
					add_core = CM1
				}
			}
			else = {
				random_province = {
					limit = {
						region = carribeans_region
						is_empty = yes
					}
					create_colony = 1000
					add_core = CM1
				}
			}
			random_province = {
				limit = {
					region = carribeans_region
					any_neighbor_province = { owned_by = CM1 }
					is_empty = yes
				}
				create_colony = 400 	
			}
		}
	}

	option = {
		name = "NHSOTHERSEVENTSNAME54"
		add_base_tax = -1
		add_base_manpower = -2
		if = {
			limit = {
				owner = {
					NOT = { num_of_colonists = 1 }
				}
			}
			owner = {
				add_country_modifier = {
					name = "nhs_new_world"
					duration = -1
				}
			}
		}
		custom_tooltip = nhs_settle_tt
	}
}
