namespace = nhs_vov_ai_help

# AI colonisation help Vinland and Norse Colonies
country_event = {
	id = nhs_vov_ai_help.1
	title = nhs_vov_ai_help.1.t
	desc = nhs_vov_ai_help.1.d
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		OR = {
			tag = MK1
			tag = HLL
			tag = VIN
		}
		ai = yes
		NOT = { num_of_colonies = 1 }
		num_of_colonists = 1
		NOT = { has_country_modifier = nhs2_ai_colonising_timer }
		OR = {
			any_province = {
				has_discovered = ROOT
				is_empty = yes
				range = ROOT
			}
			any_owned_province = {
				has_empty_adjacent_province = yes
			}
		}
	}
	
	immediate = {
		add_country_modifier = {
			name = nhs2_ai_colonising_timer
			duration = 1
			hidden = yes
		}
		if = {
			limit = {
				994 = {
					is_empty = yes
					OR = {
						range = ROOT
						any_neighbor_province = {
							owned_by = ROOT
						}
					}
				}
			}
			994 = {
				set_province_flag = vov_ai_colony_tar@ROOT
			}
		}
		else_if = {
			limit = {
				tag = MK1
				any_owned_province = {
					any_empty_neighbor_province = {
						vov_is_markland_path = yes
					}
				}
			}
			random_owned_province = {
				limit = {
					any_empty_neighbor_province = {
						vov_is_markland_path = yes
					}
				}
				random_empty_neighbor_province = {
					limit = {
						vov_is_markland_path = yes
					}
					set_province_flag = vov_ai_colony_tar@ROOT
				}
			}
		}
		else_if = {
			limit = {
				tag = HLL
				any_owned_province = {
					any_empty_neighbor_province = {
						vov_is_helluland_path = yes
					}
				}
			}
			random_owned_province = {
				limit = {
					any_empty_neighbor_province = {
						vov_is_helluland_path = yes
					}
				}
				random_empty_neighbor_province = {
					limit = {
						vov_is_helluland_path = yes
					}
					set_province_flag = vov_ai_colony_tar@ROOT
				}
			}
		}
		else_if = {
			limit = {
				tag = VIN
				any_owned_province = {
					any_empty_neighbor_province = {
						OR = {
							vov_is_markland_path = yes
							vov_is_helluland_path = yes
						}
					}
				}
			}
			random_owned_province = {
				limit = {
					any_empty_neighbor_province = {
						OR = {
							vov_is_markland_path = yes
							vov_is_helluland_path = yes
						}
					}
				}
				random_empty_neighbor_province = {
					limit = {
						OR = {
							vov_is_markland_path = yes
							vov_is_helluland_path = yes
						}
					}
					set_province_flag = vov_ai_colony_tar@ROOT
				}
			}
		}
		else_if = {
			limit = {
				tag = MK1
				any_province = {
					vov_is_markland_path = yes
					range = ROOT
					is_empty = yes
				}
			}
			random_province = {
				limit = {
					vov_is_markland_path = yes
					range = ROOT
					is_empty = yes
				}
				set_province_flag = vov_ai_colony_tar@ROOT
			}
		}
		else_if = {
			limit = {
				tag = HLL
				any_province = {
					vov_is_helluland_path = yes
					range = ROOT
					is_empty = yes
				}
			}
			random_province = {
				limit = {
					vov_is_helluland_path = yes
					range = ROOT
					is_empty = yes
				}
				set_province_flag = vov_ai_colony_tar@ROOT
			}
		}
		else_if = {
			limit = {
				tag = VIN
				any_province = {
					OR = {
						vov_is_markland_path = yes
						vov_is_helluland_path = yes
					}
					range = ROOT
					is_empty = yes
				}
			}
			random_province = {
				limit = {
					OR = {
						vov_is_markland_path = yes
						vov_is_helluland_path = yes
					}
					range = ROOT
					is_empty = yes
				}
				set_province_flag = vov_ai_colony_tar@ROOT
			}
		}
		else_if = {
			limit = {
				any_owned_province = {
					any_empty_neighbor_province = {
						to_vov_eligible_for_coastal_bonus = yes
					}
				}
			}
			random_owned_province = {
				limit = {
					any_empty_neighbor_province = {
						to_vov_eligible_for_coastal_bonus = yes
					}
				}
				random_empty_neighbor_province = {
					limit = {
						to_vov_eligible_for_coastal_bonus = yes
					}
					set_province_flag = vov_ai_colony_tar@ROOT
				}
			}
		}
		else_if = {
			limit = {
				any_province = {
					range = ROOT
					is_empty = yes
					to_vov_eligible_for_coastal_bonus = yes
				}
			}
			random_province = {
				limit = {
					range = ROOT
					is_empty = yes
					to_vov_eligible_for_coastal_bonus = yes
				}
				set_province_flag = vov_ai_colony_tar@ROOT
			}
		}
		else_if = {
			limit = {
				any_owned_province = {
					has_empty_adjacent_province = yes
				}
			}
			random_owned_province = {
				limit = {
					has_empty_adjacent_province = yes
				}
				random_empty_neighbor_province = {
					set_province_flag = vov_ai_colony_tar@ROOT
				}
			}
		}
		else_if = {
			limit = {
				any_province = {
					range = ROOT
					is_empty = yes
				}
			}
			random_province = {
				limit = {
					range = ROOT
					is_empty = yes
				}
				set_province_flag = vov_ai_colony_tar@ROOT
			}
		}
		capital_scope = {
			province_event = { id = nhs_vov_ai_help.5 }
		}
	}
	option = {
		name = "nhs_vov_ai_help.1.a"
	}
}

	id = nhs_vov_ai_help.5
	title = nhs2_hidden.t
	desc = nhs2_hidden.d
	picture = COLONIZATION_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	immediate = {
		owner = {
			save_event_target_as = vov_owner
		}
		random_province = {
			limit = {
				has_province_flag = vov_ai_colony_tar@event_target:vov_owner
			}

			clr_province_flag = vov_ai_colony_tar@event_target:vov_owner
		}
	}
	
	option = {
		name = "nhs_vov_ai_help.1.a"
	}
