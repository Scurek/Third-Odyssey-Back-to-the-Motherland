#Oracle found!
country_event = {
	id = msg_religionevents.0
	title = msg_religionevents.0.t
	desc = msg_religionevents.0.d
	picture = Oracle_eventPicture

	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = MSN
			tag = MSG
		}
		religion_group = pagan
		OR = {
			had_country_flag = { flag = nhs_msg_land1 days = 3650 }
			AND = {
				had_country_flag = { flag = nhs_msg_land1 days = 1825 }
				has_global_flag = nhs_started_at_bookmark
			}
		}
	}
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.1
			has_global_flag = nhs_started_at_bookmark
		}
	}
	
	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = nhs2_oracle_cooldown
				duration = 3650
				hidden = yes
			}
		}
	}
	
	option = {
		name = msg_religionevents.0.a
		add_country_modifier = {
			name = "msg_oracle_before"
			duration = -1
		}
	}
}
#Oracle envisions the rise of the great serpent
country_event = {
	id = msg_religionevents.1
	title = msg_religionevents.1.t
	desc = msg_religionevents.1.d
	picture = Oracle_eventPicture

	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = MSN
			tag = MSG
		}
		religion_group = pagan
		has_country_modifier = msg_oracle_before
		has_country_modifier = nhs_high_priests
		OR = {
			had_country_flag = { flag = nhs_msg_land1 days = 7300 }
			AND = {
				had_country_flag = { flag = nhs_msg_land1 days = 5475 }
				has_global_flag = nhs_started_at_bookmark
			}
		}
		NOT = { has_country_flag = aztlan_reformation }
		NOT = { has_global_flag = msg_post_reformation }
	}
	mean_time_to_happen = {
		months = 5 
	}
	option = {
		name = msg_religionevents.1.a
		if = {
			limit = {
				OR = {
					NOT = {	has_country_modifier = nhs_expanded_pantheon }
					NOT = {	has_country_modifier = nhs_consolidated_pantheon }
				}	
			}
			add_country_modifier = {
				name = nhs_consolidated_pantheon
				duration = -1
			}
		}
		every_owned_province = {
			limit = {
				has_owner_religion = no
			}
			add_unrest = 5
		}
	}
	option = {
		name = msg_religionevents.1.b
		if = {
			limit = {
				OR = {
					NOT = {	has_country_modifier = nhs_expanded_pantheon }
					NOT = {	has_country_modifier = nhs_consolidated_pantheon }
				}	
			}
			add_country_modifier = {
				name = nhs_expanded_pantheon
				duration = -1
			}
		}
		every_owned_province = {
			limit = {
				has_owner_religion = yes
			}
			add_unrest = 5
		}
	}
	option = {
		name = msg_religionevents.1.c
		add_prestige = 50
		add_stability = -1
	}
	
}

	desc = msg_religionevents.2.d

	fire_only_once = yes
	
	trigger = {
		owned_by = MSN
		MSN = {
			religion_group = pagan
		}
		owner = { 
			OR = {
				had_country_flag = { flag = nhs_msg_land1 days = 18250 }
				AND = {
					had_country_flag = { flag = nhs_msg_land1 days = 16425 }
					has_global_flag = nhs_started_at_bookmark
				}
				AND = {
					has_country_flag = nhs_msg_land3
					is_year = 1520
				}
				is_year = 1540
			}
		}
		religion_group = pagan
		OR = {
			area = gran_chichimeca_area
			area = jalisco_area
			area = oaxaca_area
			area = guerrero_area
			area = michoacan_area
			area = mexico_area
			area = eastern_mexico_area
			region = central_america_region
		}
		NOT = {
			is_religion_enabled = aztlan_worship
			province_id = 842
			province_id = 846
			province_id = 848
			province_id = 852
		}
		OR = {
			AND = {
				religion = pantheon_worship
				any_neighbor_province = {
					OR = {
						religion = mesoamerican_religion
						religion = nahuatl
					}
				}
			}
			AND = {
				OR = {
						religion = mesoamerican_religion
						religion = nahuatl
					}
				any_neighbor_province = {
					religion = pantheon_worship
				}
			}
			owner = {
				NOT = {
					any_owned_province = {
						NOT = { religion = ROOT }
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 120
		
		modifier = {
			factor = 0.9
			owner = { innovativeness_ideas = 5 }
		}
		modifier = {
			factor = 0.9
			owner = { innovativeness_ideas = 6 }
		}
		modifier = {
			factor = 0.9
			owner = { innovativeness_ideas = 7 }
		}
		modifier = {
			factor = 0.9
			owner = { has_country_flag = nhs_allows_polygamy }
		}
		modifier = {
			factor = 0.9
			owner = { has_country_flag = nhs_high_priests }
		}
		modifier = {
			factor = 1.5
			OR = { 
				owner = { has_country_modifier = nhs_ares_worship }
				owner = { has_country_modifier = nhs_poseidon_worship }
				}
		}
		modifier = {
			factor = 0.8
			owner = { has_idea = humanist_tolerance }
		}
		modifier = {
			factor = 0.95
			owner = { NOT = { nhs_ely_has_opinion_X = { MNENJE = -50 } } }
		}
		modifier = {
			factor = 0.85
			owner = { NOT = { nhs_ely_has_opinion_X = { MNENJE = -100 } } }
		}
		modifier = {
			factor = 0.75
			owner = { NOT = { nhs_ely_has_opinion_X = { MNENJE = -150 } } }
		}	
		modifier = {
			factor = 0.5
			owner = { num_of_revolts = 1 }
		}
		modifier = {
			factor = 0.5
			OR = {
				AND = {
					religion = pantheon_worship
					any_neighbor_province = {
						OR = {
							religion = mesoamerican_religion
							religion = nahuatl
						}
					}
				}
				AND = {
					OR = {
							religion = mesoamerican_religion
							religion = nahuatl
						}
					any_neighbor_province = {
						religion = pantheon_worship
					}
				}
			}
		}
	}
	
	immediate = {
		set_global_flag = aztlan_reformation
	}
	
	option = {
		name = msg_religionevents.2.a
		enable_religion = aztlan_worship
		change_religion = aztlan_worship

			name = religious_center
			duration = -1
		}
		add_permanent_province_modifier = {
			name = religious_zeal_at_conv
			duration = 9000
		}
		owner = {
			remove_country_modifier = msg_oracle_before
			swap_non_generic_missions = yes
		}
	}
}
