#Norse to Vinlandic
country_event = {
	id = nhs_vov_culture.1
	title = nhs_vov_culture.1.t
	desc = nhs_vov_culture.1.d
	picture = norse_district_eventPicture
	
	trigger = {
		primary_culture = norse
		is_year = 1470
		capital_scope = {
			continent = north_america
		}
	}
	mean_time_to_happen = {
		months = 1
		modifier = {
			factor = 0.000001
			has_global_flag = nhs_vov_vinland_culture
		}
	}
	option = {
		name = "nhs_vov_culture.1.a"
		change_primary_culture = vinlandic
		if = {
			limit = {
				ruler_culture = norse
			}
			set_ruler_culture = vinlandic
		}
		if = {
			limit = {
				consort_culture = norse
				has_consort = yes
			}
			set_consort_culture = vinlandic
		}
		if = {
			limit = {
				heir_culture = norse
				has_heir = yes
			}
			set_heir_culture = vinlandic
		}
		hidden_effect = {
			every_province = {
				limit = {
					culture = norse
					OR = {
						owned_by = ROOT
						owner = {
							NOT = { culture_group = norse }
						}
					}
				}
				change_culture = vinlandic
				hidden_effect = {
					nhs_on_province_culture_converted_general = yes
				}
			}
		}
		set_global_flag = nhs_vov_vinland_culture
	}
}

#Native american to Skraeling
country_event = {
	id = nhs_vov_culture.2
	title = nhs_vov_culture.2.t
	desc = nhs_vov_culture.2.d
	picture = norse_district_eventPicture
	
	trigger = {
		culture_group = norse_g
		NOT = { primary_culture = norse }
		any_owned_province = {
			to_vov_can_convert_to_skraeling = yes
		}
	}

	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					to_vov_can_convert_to_skraeling = yes
				}
				save_event_target_as = nhs_vov_culture2_target
			}
		}
	}

	mean_time_to_happen = {
		months = 90
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 30
				}
			}
		}
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 20
				}
			}
		}
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 10
				}
			}
		}
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 1
				}
			}
		}
		modifier = {
			factor = 0.5
			has_estate_privilege = to_vov_native_priv_benevolent_conquerors
		}
		modifier = {
			factor = 0.5
			has_estate_privilege = to_vov_native_priv_open_ranks
		}
		modifier = {
			factor = 0.9
			stability = 1
		}
		modifier = {
			factor = 0.9
			stability = 2
		}
		modifier = {
			factor = 0.9
			stability = 3
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 10
				to_vov_can_convert_to_skraeling = yes
			}
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 15
				to_vov_can_convert_to_skraeling = yes
			}
		}
	}

	option = {		# Great news
		name = "nhs_vov_culture.2.a"
		goto = nhs_vov_culture2_target
		event_target:nhs_vov_culture2_target = {
			if = {
				limit = {
					native_culture = yes
					NOT = { culture = skraeling }
				}
				change_culture = skraeling
			}
		}
		event_target:nhs_vov_culture2_target = {
			to_nat_assimilate_local_tribal_development = yes
		}
		hidden_effect = {
			event_target:nhs_vov_culture2_target = {
				clr_province_flag = nhs_culture_changed_to_skraeling
				set_province_flag = nhs_culture_changed_to_skraeling
				hidden_effect = {
					nhs_on_province_culture_converted_general = yes
				}
			}
		}
	}
}

#Vinlandic to Vatnmadr
country_event = {
	id = nhs_vov_culture.6
	title = nhs_vov_culture.6.t
	desc = nhs_vov_culture.6.d
	picture = niagara_falls_eventPicture

	trigger = {
		culture_group = norse_g
		NOT = { primary_culture = norse }
		NOT = { has_country_flag = to_vin_vatnmadr_culture }
		any_owned_province = {
			is_colony = no
			culture_group = norse_g
			region = great_lakes_region
			NOT = { culture = norse }
			NOT = { culture = skraeling }
		}
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.001
			any_owned_province = {
				is_colony = no
				culture = vatnmadr
			}
		}
	}
	
	option = {
		name = "nhs_vov_culture.6.a"
		if = {
			limit = {
				NOT = { nhs_check_all_norse = { CONDITION = tag } }
				capital_scope = {
					region = great_lakes_region
				}
			}
			change_primary_culture = vatnmadr
		}
		custom_tooltip = to_vin_spawn_vatnmadr_tt
		hidden_effect = {
			every_province = {
				limit = {
					culture_group = norse_g
					region = great_lakes_region
					NOT = { culture = norse }
					NOT = { culture = vatnmadr }
					NOT = { culture = skraeling }
					OR = {
						owned_by = ROOT
						owner = {
							NOT = { culture_group = norse_g }
						}
					}
				}
				change_culture = vatnmadr
				hidden_effect = {
					nhs_on_province_culture_converted_general = yes
				}
			}
			set_country_flag = to_vin_vatnmadr_culture
		}
	}
}
