# Cultural change for Rhomania - Greek to romaic
country_event = {
	id = nhs2_cultureevents.100
	title = nhs2_cultureevents.100.t
	desc = nhs2_cultureevents.100.d
	picture = byz_COURT_eventPicture

	fire_only_once = yes
	
	trigger = {
		nhs_check_all_elysian_tags = { CONDITION = tag }
		is_year = 1500
		primary_culture = greek
		NOT = { has_global_flag = nhs_romaic_culture_change }
		OR = {
			capital_scope = { continent = north_america }
			capital_scope = { continent = south_america }
		}
		technology_group = exiled_greek
		OR = {
			is_year = 1525
			AND = {
				is_year = 1515
				NOT = {
					to_ely_has_integrating_privilege = yes
				}
				calc_true_if = {
					amount = 10
					all_owned_province = {
						OR = {
							culture = elysameric
							has_province_flag = to_culture_changed_from_symmachoi
							to_ely_can_convert_to_barbaroi = yes
						}
					}
				}
			}
			AND = {
				is_year = 1515
				to_ely_has_integrating_privilege = yes
				any_owned_province = {
					OR = {
						culture = elysameric
						has_province_flag = to_culture_changed_from_symmachoi
						to_ely_can_convert_to_barbaroi = yes
					}
				}
			}
			AND = {
				is_year = 1505
				to_ely_has_integrating_privilege = yes
				calc_true_if = {
					amount = 10
					all_owned_province = {
						OR = {
							culture = elysameric
							has_province_flag = to_culture_changed_from_symmachoi
							to_ely_can_convert_to_barbaroi = yes
						}
					}
				}
			}
			AND = {
				is_year = 1490
				to_ely_has_integrating_privilege = yes
				calc_true_if = {
					amount = 20
					all_owned_province = {
						OR = {
							culture = elysameric
							has_province_flag = to_culture_changed_from_symmachoi
							to_ely_can_convert_to_barbaroi = yes
						}
					}
				}
			}
		}
	}

	immediate = {
		set_global_flag = nhs_romaic_culture_change
	}

	option = {		# What will be, will be
		name = nhs2_cultureevents.100.a
		tooltip = {
			change_primary_culture = romaic
		}
		if = {
			limit = {
				has_country_flag = nhs_enable_venetian_merchant_estate
			}
			custom_tooltip = nhs_rename_merchants_back_tt
		}
		hidden_effect = {
			if = {
				limit = {
					ruler_culture = greek
				}
				set_ruler_culture = romaic
			}
			if = {
				limit = {
					consort_culture = greek
					has_consort = yes
				}
				set_consort_culture = romaic
			}
			if = {
				limit = {
					heir_culture = greek
					has_heir = yes
				}
				set_heir_culture = romaic
			}
		}
		hidden_effect = {
			every_province = {
				limit = {
					owned_by = ROOT
					NOT = { culture = greek }
					has_construction = culture
					NOT = { has_albanian_majority_modifier = yes }
				}
				set_province_flag = to_check_culture_after_conversion
			}
			change_primary_culture = romaic
			every_province = {
				limit = {
					culture = greek
					OR = {
						AND = {
							owned_by = ROOT
							NOT = { continent = europe }
						}
						continent = north_america
						continent = south_america
					}
					NOT = { has_albanian_majority_modifier = yes }
					NOT = {
						owner = {
							primary_culture = greek
						}
					}
				}
				change_culture = romaic
			}
			every_country = {
				country_event = { id = nhs2_cultureevents.106 }
				country_event = { id = nhs2_cultureevents.107 }
			}
		}
	}
}

# Appearance of Symmachoi via cultural assimilation
province_event = {
	id = nhs2_cultureevents.201
	title = nhs2_cultureevents.201.t
	desc = nhs2_cultureevents.201.d
	picture = nat_COURT_eventPicture
	
	is_triggered_only = yes

	goto = ROOT

	immediate = {
		hidden_effect = {
			owner = {
				set_country_flag = nhs_barbaroi_integration
			}
			if = {
				limit = {
					NOT = { has_global_flag = nhs_first_barbaroi_integration }
				}
				set_global_flag = nhs_first_barbaroi_integration
			}
		}
	}

	option = {		# What will be, will be
		name = nhs2_cultureevents.201.a
		if = {
			limit = {
				nhs_culturally_native = yes
			}
			change_culture = elysameric
		}
		to_nhs_handle_post_disaster_tolerance_modifier_culture_integration = yes
		to_nat_assimilate_local_tribal_development = yes
		if = {
			limit = {
				owner = {
					NOT = { has_country_flag = to_assimilation_notifications_potential }
				}
			}
			custom_tooltip = nhs_new_line_tt
			custom_tooltip = to_enable_assimilation_notifications_toggle_tt
			hidden_effect = {
				owner = {
					set_country_flag = to_assimilation_notifications_potential
				}
			}
		}
		if = {
			limit = {
				owner = {
					OR = {
						has_government_mechanic = to_roman_janissary_mechanic
						AND = {
							is_colonial_nation = no
							NOT = { is_subject_of_type = elysian_subject_varangian }
							NOT = { has_country_flag = native_council_enabled_mid }
						}
					}
				}
			}
			custom_tooltip = nhs_dotted_line_tt
			if = {
				limit = {
					owner = {
						has_government_mechanic = to_roman_janissary_mechanic
					}
				}
				custom_tooltip = to_modify_from_true_faith_symmachoi_tt
				custom_tooltip = to_from_true_faith_effect_symmachoi_tt
				custom_tooltip = nhs_new_line_tt
			}
			if = {
				limit = {
					owner = {
						is_colonial_nation = no
						NOT = { is_subject_of_type = elysian_subject_varangian }
						NOT = { has_country_flag = native_council_enabled_mid }
					}
				}
				to_switch_elysian_native_estate_to_mid = { SCOPE = owner }
			}
		}
	}
}

# Appearance of Symmachoi via province ownership
country_event = {
	id = nhs2_cultureevents.202
	title = nhs2_cultureevents.201.t
	desc = nhs2_cultureevents.201.d
	picture = nat_COURT_eventPicture
	
	trigger = {
		culture_group = byzantine
		# nhs_check_all_elysian_tags_expanded = { CONDITION = tag }
		NOT = { is_subject_of_type = elysian_subject_varangian }
		NOT = { has_country_flag = nhs_barbaroi_integration }
		if = {
			limit = {
				is_colonial_nation = no
			}
			OR = {
				has_country_flag = native_council_enabled_early
				has_country_flag = native_council_enabled_mid
			}
		}
		else = {
			overlord = {
				OR = {
					has_country_flag = native_council_enabled_early
					has_country_flag = native_council_enabled_mid
				}
			}
		}
		any_owned_province = {
			culture = elysameric
		}
	}

	mean_time_to_happen = {
		months = 3
	}

	immediate = {
		hidden_effect = {
			set_country_flag = nhs_barbaroi_integration
			if = {
				limit = {
					NOT = { has_global_flag = nhs_first_barbaroi_integration }
				}
				set_global_flag = nhs_first_barbaroi_integration
			}
		}
	}

	option = {		# What will be, will be
		name = nhs2_cultureevents.201.a
		if = {
			limit = {
				has_government_mechanic = to_roman_janissary_mechanic
			}
			custom_tooltip = to_modify_from_true_faith_symmachoi_tt
			custom_tooltip = to_from_true_faith_effect_symmachoi_tt
			custom_tooltip = nhs_new_line_tt
		}
		if = {
			limit = {
				is_colonial_nation = no
				NOT = { is_subject_of_type = elysian_subject_varangian }
				NOT = { has_country_flag = native_council_enabled_mid }
			}
			to_switch_elysian_native_estate_to_mid = { SCOPE = ROOT }
		}
	}
}

# Greek -> romaic
country_event = {
	id = nhs2_cultureevents.106
	title = nhs2_cultureevents.106.t
	desc = nhs2_cultureevents.106.d
	picture = COLONIZATION_eventPicture

	trigger = {
		NOT = { nhs_check_all_elysian_tags = { CONDITION = tag } }
		NOT = { tag = CM1 }
		primary_culture = greek
		technology_group = exiled_greek
		OR = {
			capital_scope = { continent = north_america }
			capital_scope = { continent = south_america }
		}
		OR = {
			has_global_flag = nhs_romaic_to_elysian
			has_global_flag = nhs_romaic_culture_change
		}
	}

	option = {
		name = "nhs2_cultureevents.106.a"
		hidden_effect = {
			every_province = {
				limit = {
					owned_by = ROOT
					NOT = { culture = ROOT }
					has_construction = culture
					NOT = { has_albanian_majority_modifier = yes }
				}
				set_province_flag = to_check_culture_after_conversion
			}
		}
		if = {
			limit = {
				is_subject = yes
				overlord = {
					primary_culture = elysian
				}
			}
			to_primary_culture_and_ruler_heir_consort_culture_if_primary = {
				CULTURE = elysian
				PRIMARY_CULTURE = THIS
			}
			hidden_effect = {
				every_province = {
					limit = {
						owned_by = ROOT
						culture = greek
						NOT = { continent = europe }
					}
					change_culture = elysian
				}
			}
		}
		else = {
			to_primary_culture_and_ruler_heir_consort_culture_if_primary = {
				CULTURE = romaic
				PRIMARY_CULTURE = THIS
			}
			hidden_effect = {
				every_province = {
					limit = {
						owned_by = ROOT
						culture = greek
						NOT = { continent = europe }
					}
					change_culture = romaic
				}
			}
		}
	}
}

# Borealians appear
country_event = {
	id = nhs2_cultureevents.8
	title = nhs2_cultureevents.8.t
	desc = nhs2_cultureevents.8.d
	picture = niagara_falls_eventPicture
	
    trigger = {
		nhs_post_fragmentation_culturally_elysian = yes
		NOT = { has_country_flag = nhs_borealians_appeared_country }
		to_can_spawn_roman_subculture = {
			GLOBAL_FLAG = nhs_borealians_appeared
			COUNTRY_FLAG = nhs_borealians_appear
			PROVINCE_TRIGGER = "nhs_borealian_areas = yes"
		}
	}

	mean_time_to_happen = {
		months = 3
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = nhs_borealians_appeared
			set_country_flag = nhs_borealians_appeared_country
		}
	}

	option = {
		name = "nhs2_cultureevents.8.a"
		if = {
			limit = {
				NOT = { nhs_check_all_elysian_tags = { CONDITION = tag } }
				capital_scope = {
					nhs_borealian_areas = yes
				}
			}
			hidden_effect = {
				every_province = {
					limit = {
						owned_by = ROOT
						NOT = { culture = ROOT }
						has_construction = culture
						NOT = { has_albanian_majority_modifier = yes }
					}
					set_province_flag = to_check_culture_after_conversion
				}
			}
			to_primary_culture_and_ruler_heir_consort_culture_if_primary = {
				CULTURE = borealian
				PRIMARY_CULTURE = THIS
			}
		}
		if = {
			limit = {
				NOT = { has_country_flag = nhs_hyperborealians_appeared_country }
				any_owned_province = {
					nhs_hyperborealian_areas = yes
				}
			}
			custom_tooltip = nhs_spawn_borealian_additional_tt
		}
		else = {
			custom_tooltip = nhs_spawn_borealian_tt
		}
		hidden_effect = {
			every_province = {
				limit = {
					owner = {
						OR = {
							tag = ROOT
							NOT = { culture_group = byzantine }
						}
					}
					to_nhs_has_dynamic_elysian_culture = yes
					OR = {
						nhs_borealian_areas = yes
						AND = {
							owner = {
								if = {
									limit = {
										culture_group = byzantine
									}
									NOT = { has_country_flag = nhs_hyperborealians_appeared_country }
								}
								else = {
									NOT = { has_global_flag = nhs_hyperborealians_appeared }
								}
							}
							nhs_hyperborealian_areas = yes
						}	
					}
					NOT = { culture = borealian }
				}
				change_culture = borealian
			}
			every_province = {
				limit = {
					owned_by = ROOT
					has_construction = culture
					OR = {
						nhs_borealian_areas = yes
						AND = {
							owner = {
								if = {
									limit = {
										culture_group = byzantine
									}
									NOT = { has_country_flag = nhs_hyperborealians_appeared_country }
								}
								else = {
									NOT = { has_global_flag = nhs_hyperborealians_appeared }
								}
