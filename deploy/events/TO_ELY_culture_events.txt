# Cultural change for Rhomania - Greek to romaic
country_event = {
	id = nhs2_cultureevents.100
	title = nhs2_cultureevents.100.t
	desc = nhs2_cultureevents.100.d
	picture = byz_COURT_eventPicture

	fire_only_once = yes
	
	trigger = {
		nhs_condition_all_ely = { KOMANDA = tag }
		is_year = 1500
		primary_culture = greek
		NOT = { has_global_flag = nhs_romaic_culture_change }
		OR = {
			capital_scope = { continent = north_america }
			capital_scope = { continent = south_america }
		}
		technology_group = exiled_greek
		OR = {
			AND = {
				is_year = 1515
				NOT = {
					to_ely_has_integrating_privilege = yes
				}
				calc_true_if = {
					amount = 10
					all_owned_province = {
						OR = {
							culture = elysameric
							to_ely_can_convert_to_barbaroi = yes
						}
					}
				}
			}
			AND = {
				is_year = 1515
				to_ely_has_integrating_privilege = yes
				any_owned_province = {
					OR = {
						culture = elysameric
						to_ely_can_convert_to_barbaroi = yes
					}
				}
			}
			AND = {
				is_year = 1505
				to_ely_has_integrating_privilege = yes
				calc_true_if = {
					amount = 10
					all_owned_province = {
						OR = {
							culture = elysameric
							to_ely_can_convert_to_barbaroi = yes
						}
					}
				}
			}
			AND = {
				is_year = 1490
				to_ely_has_integrating_privilege = yes
				calc_true_if = {
					amount = 20
					all_owned_province = {
						OR = {
							culture = elysameric
							to_ely_can_convert_to_barbaroi = yes
						}
					}
				}
			}
		}
	}

	immediate = {
		set_global_flag = nhs_romaic_culture_change
	}

	option = {		# What will be, will be
		name = nhs2_cultureevents.100.a
		tooltip = {
			change_primary_culture = romaic
		}
		if = {
			limit = {
				has_country_flag = nhs_enable_venetian_merchant_estate
			}
			custom_tooltip = nhs_rename_merchants_back_tt
		}
		hidden_effect = {
			if = {
				limit = {
					ruler_culture = greek
				}
				set_ruler_culture = romaic
			}
			if = {
				limit = {
					consort_culture = greek
					has_consort = yes
				}
				set_consort_culture = romaic
			}
			if = {
				limit = {
					heir_culture = greek
					has_heir = yes
				}
				set_heir_culture = romaic
			}
		}
		hidden_effect = {
			every_owned_province = {
				limit = {
					culture = greek
					NOT = { continent = europe }
					NOT = { has_albanian_majority_modifier = yes }
				}
				change_culture = romaic
				hidden_effect = {
					nhs_on_province_culture_converted_general = yes
				}
			}
			every_province = {
				limit = {
					OR = {
						continent = north_america
						continent = south_america
					}
					culture = greek
					NOT = {
						owner = {
							primary_culture = greek
						}
					}
				}
				change_culture = romaic
				hidden_effect = {
					nhs_on_province_culture_converted_general = yes
				}
			}
			change_primary_culture = romaic
			every_country = {
				country_event = { id = nhs2_cultureevents.106 }
				country_event = { id = nhs2_cultureevents.107 }
			}
		}
	}
}

# Appearance of Symmachoi
country_event = {
	id = nhs2_cultureevents.201
	title = nhs2_cultureevents.201.t
	desc = nhs2_cultureevents.201.d
	picture = byz_COURT_eventPicture
	
	trigger = {
		culture_group = byzantine
		nhs_check_all_elysian_expanded = { CONDITION = tag }
		NOT = { is_subject_of_type = elysian_subject_varangian }
		NOT = { has_country_flag = nhs_barbaroi_integration }
		if = {
			limit = {
				is_colonial_nation = no
			}
			OR = {
				has_country_flag = native_council_enabled_early
				has_country_flag = native_council_enabled_mid
			}
		}
		else = {
			overlord = {
				OR = {
					has_country_flag = native_council_enabled_early
					has_country_flag = native_council_enabled_mid
				}
			}
		}
		any_owned_province = {
			culture = elysameric
		}
	}

	mean_time_to_happen = {
		months = 3
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					NOT = { has_global_flag = nhs_first_barbaroi_integration }
				}
				set_global_flag = nhs_first_barbaroi_integration
			}
		}
	}

	option = {		# What will be, will be
		name = nhs2_cultureevents.201.a
		set_country_flag = nhs_barbaroi_integration
		if = {
			limit = {
				NOT = { has_country_flag = native_council_enabled_mid }
			}
			nhs_switch_elysian_native_estate_to_mid = yes
		}
	}
}

# Gradual cultural conversion of native american to Symmachoi
country_event = {
	id = nhs2_cultureevents.102
	title = nhs2_cultureevents.102.t
	desc = nhs2_cultureevents.102.d
	picture = NATIVES_eventPicture
	
	trigger = {
		culture_group = elysian_group
		has_country_flag = nhs_barbaroi_integration
		any_owned_province = {
			to_ely_can_convert_to_barbaroi = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			random_list = {
				80 = {
					trigger = {
						any_owned_province = {
							to_ely_can_convert_to_barbaroi = yes
							OR = {
								has_port = yes
								any_neighbor_province = {
									culture_group = ROOT
								}
							}
						}
					}
					random_owned_province = {
						limit = {
							to_ely_can_convert_to_barbaroi = yes
							OR = {
								has_port = yes
								any_neighbor_province = {
									culture_group = ROOT
								}
							}
						}
						save_event_target_as = nhs_countryevents15_target
					}
				}
				20 = {
					random_owned_province = {
						limit = {
							to_ely_can_convert_to_barbaroi = yes
						}
						save_event_target_as = nhs_countryevents15_target
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 90
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 30
				}
			}
		}
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 20
				}
			}
		}
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 10
				}
			}
		}
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 1
				}
			}
		}
		modifier = {
			factor = 0.8
			has_estate_privilege = nhs_native_priv_citizenship_for_service
		}
		modifier = {
			factor = 0.5
			OR = {
				has_estate_privilege = nhs_native_priv_encourage_barbaroi_officials
				has_estate_privilege = to_vov_native_priv_open_ranks
			}
		}
		modifier = {
			factor = 0.5
			OR = {
				has_estate_privilege = nhs_native_priv_inclusive_citizenship_mid
				has_estate_privilege = to_vov_native_priv_benevolent_conquerors
			}
		}
		modifier = {
			factor = 0.8
			has_active_policy = elysian_integration_act
		}
		modifier = {
			factor = 0.9
			stability = 1
		}
		modifier = {
			factor = 0.9
			stability = 2
		}
		modifier = {
			factor = 0.9
			stability = 3
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 10
				to_ely_can_convert_to_barbaroi = yes
			}
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 15
				to_ely_can_convert_to_barbaroi = yes
			}
		}
		# modifier = {
		# 	factor = 1.5
		# 	NOT = { has_country_modifier = nhs_extended_city_senate }
		# }
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					NOT = { has_country_flag = nhs_barbaroi_integration }
				}
				set_country_flag = nhs_barbaroi_integration
			}
		}
	}

	option = {	# Great news
		name = "nhs2_cultureevents.102.a"
		goto = nhs_countryevents15_target
		event_target:nhs_countryevents15_target = {
			if = {
				limit = {
					nhs_culturally_native = yes
				}
				change_culture = elysameric
			}
		}
		event_target:nhs_countryevents15_target = {
			to_nhs_handle_post_disaster_tolerance_modifier_culture_integration = yes
		}
		event_target:nhs_countryevents15_target = {
			to_nat_assimilate_local_tribal_development = yes
		}
		hidden_effect = {
			event_target:nhs_countryevents15_target = {
				set_province_flag = nhs_culture_changed_to_symmachoi
				hidden_effect = {
					nhs_on_province_culture_converted_general = yes
				}
			}
		}
		
		if = {
			limit = {
				NOT = { has_country_flag = native_council_enabled_mid }
			}
			custom_tooltip = nhs_dotted_line_tt
			nhs_switch_elysian_native_estate_to_mid = yes
		}
	}
}
