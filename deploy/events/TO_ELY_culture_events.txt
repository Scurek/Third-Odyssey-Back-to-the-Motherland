# Cultural change for Elysia - Greek to Elysian, native influences
country_event = {
	id = nhs2_cultureevents.100
	title = nhs2_cultureevents.100.t
	desc = nhs2_cultureevents.100.d
	picture = byz_COURT_eventPicture

	fire_only_once = yes
	
	trigger = {
		nhs_condition_all_ely = { KOMANDA = tag }
		is_year = 1500
		primary_culture = greek
		NOT = { has_global_flag = nhs_elysian_culture_change }
		OR = {
			capital_scope = { continent = north_america }
			capital_scope = { continent = south_america }
		}
		OR = {
			technology_group = exiled_greek
			technology_group = modern_greek
		}
		OR = {
			AND = {
				is_year = 1515
				NOT = {
					to_ely_has_integrating_privilege = yes
				}
				calc_true_if = {
					amount = 10
					all_owned_province = {
						to_ely_can_convert_to_barbaroi = yes
					}
				}
			}
			AND = {
				is_year = 1515
				to_ely_has_integrating_privilege = yes
				any_owned_province = {
					to_ely_can_convert_to_barbaroi = yes
				}
			}
			AND = {
				is_year = 1505
				to_ely_has_integrating_privilege = yes
				calc_true_if = {
					amount = 10
					all_owned_province = {
						to_ely_can_convert_to_barbaroi = yes
					}
				}
			}
			AND = {
				is_year = 1490
				to_ely_has_integrating_privilege = yes
				calc_true_if = {
					amount = 20
					all_owned_province = {
						to_ely_can_convert_to_barbaroi = yes
					}
				}
			}
		}
	}

	immediate = {
		set_global_flag = nhs_elysian_culture_change
		set_global_flag = nhs_first_barbaroi_integration
		hidden_effect = {
			save_event_target_as = previous_tag
		}
	}

	option = {		# What will be, will be
		name = nhs2_cultureevents.100.a
		change_tag = ELY
		to_restore_historal_friends_and_rivals = { PREVIOUS_TAG = event_target:previous_tag }
		tooltip = {
			change_primary_culture = elysian
		}
		if = {
			limit = {
				has_country_flag = nhs_enable_venetian_merchant_estate
			}
			custom_tooltip = nhs_rename_merchants_back_tt
		}
		hidden_effect = {
			if = {
				limit = {
					ruler_culture = greek
				}
				set_ruler_culture = elysian
			}
			if = {
				limit = {
					consort_culture = greek
					has_consort = yes
				}
				set_consort_culture = elysian
			}
			if = {
				limit = {
					heir_culture = greek
					has_heir = yes
				}
				set_heir_culture = elysian
			}
		}
		set_country_flag = nhs_barbaroi_integration
		custom_tooltip = nhs_convert_culture_capital_with_assimilation_tt
		custom_tooltip = nhs_new_line_tt
		nhs_switch_elysian_native_estate_to_mid = yes
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						is_capital = yes
						nhs_homelands = yes
					}
					has_province_modifier = to_nat_tribal_development
				}
				if = {
					limit = {
						nhs_culturally_native = yes
						NOT = { has_albanian_majority_modifier = yes }
					}
					change_culture = elysian
					change_religion = ROOT
				}
				to_nat_assimilate_local_tribal_development = yes
				hidden_effect = {
					nhs_on_province_culture_converted_general = yes
				}
			}
			every_owned_province = {
				limit = {
					culture = greek
					NOT = { continent = europe }
					NOT = { has_albanian_majority_modifier = yes }
				}
				change_culture = elysian
				hidden_effect = {
					nhs_on_province_culture_converted_general = yes
				}
			}
			every_province = {
				limit = {
					OR = {
						continent = north_america
						continent = south_america
					}
					culture = greek
					NOT = {
						owner = {
							primary_culture = greek
						}
					}
				}
				change_culture = elysian
				hidden_effect = {
					nhs_on_province_culture_converted_general = yes
				}
			}
			change_primary_culture = elysian
			every_country = {
				country_event = { id = nhs2_cultureevents.106 }
				country_event = { id = nhs2_cultureevents.107 }
			}
		}
	}
}

# Appearance of Symmachoi
country_event = {
	id = nhs2_cultureevents.201
	title = nhs2_cultureevents.201.t
	desc = nhs2_cultureevents.201.d
	picture = byz_COURT_eventPicture
	
	trigger = {
		culture_group = elysian_group
		nhs_check_all_elysian_expanded = { CONDITION = tag }
		NOT = { is_subject_of_type = elysian_subject_varangian }
		if = {
			limit = {
				is_colonial_nation = no
			}
			OR = {
				has_country_flag = native_council_enabled_early
				has_country_flag = native_council_enabled_mid
			}
		}
		else = {
			overlord = {
				OR = {
					has_country_flag = native_council_enabled_early
					has_country_flag = native_council_enabled_mid
				}
			}
		}
		NOT = { has_country_flag = nhs_barbaroi_integration }
		had_global_flag = {
			flag = nhs_elysian_culture_change
			days = 7300
		}
		OR = {
			AND = {
				has_global_flag = nhs_first_barbaroi_integration
				any_owned_province = {
					OR = {
						culture = elysameric
						to_ely_can_convert_to_barbaroi = yes
					}
				}
			}
			calc_true_if = {
				amount = 10
				all_owned_province = {
					to_ely_can_convert_to_barbaroi = yes
				}
			}
		}
	}

	immediate = {
		set_global_flag = nhs_first_barbaroi_integration
	}

	option = {		# What will be, will be
		name = nhs2_cultureevents.201.a
		set_country_flag = nhs_barbaroi_integration
		if = {
			limit = {
				nhs_condition_all_ely = { KOMANDA = tag }
				any_owned_province = {
					OR = {
						is_capital = yes
						nhs_homelands = yes
					}
					has_province_modifier = to_nat_tribal_development
				}
			}
			hidden_effect = {
				every_owned_province = {
					limit = {
						OR = {
							is_capital = yes
							nhs_homelands = yes
						}
						has_province_modifier = to_nat_tribal_development
					}
					if = {
						limit = {
							nhs_culturally_native = yes
							NOT = { has_albanian_majority_modifier = yes }
						}
						change_culture = elysian
						change_religion = ROOT
					}
					to_nat_assimilate_local_tribal_development = yes
					hidden_effect = {
						nhs_on_province_culture_converted_general = yes
					}
				}
			}
			custom_tooltip = nhs_convert_culture_capital_with_assimilation_tt
			custom_tooltip = nhs_new_line_tt
		}
		if = {
			limit = {
				NOT = { has_country_flag = native_council_enabled_mid }
			}
			nhs_switch_elysian_native_estate_to_mid = yes
		}
	}
}

# Gradual cultural conversion of native american to Symmachoi
country_event = {
	id = nhs2_cultureevents.102
	title = nhs2_cultureevents.102.t
	desc = nhs2_cultureevents.102.d
	picture = NATIVES_eventPicture
	
	trigger = {
		culture_group = elysian_group
		has_country_flag = nhs_barbaroi_integration
		any_owned_province = {
			to_ely_can_convert_to_barbaroi = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			random_list = {
				80 = {
					trigger = {
						any_owned_province = {
							to_ely_can_convert_to_barbaroi = yes
							OR = {
								has_port = yes
								any_neighbor_province = {
									culture_group = ROOT
								}
							}
						}
					}
					random_owned_province = {
						limit = {
							to_ely_can_convert_to_barbaroi = yes
							OR = {
								has_port = yes
								any_neighbor_province = {
									culture_group = ROOT
								}
							}
						}
						save_event_target_as = nhs_countryevents15_target
					}
				}
				20 = {
					random_owned_province = {
						limit = {
							to_ely_can_convert_to_barbaroi = yes
						}
						save_event_target_as = nhs_countryevents15_target
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 90
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 30
				}
			}
		}

			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 10
				}
			}
		}
		modifier = {
			factor = 1.5
			has_estate = estate_all_natives
			NOT = {
				estate_loyalty = {
					estate = estate_all_natives
					loyalty = 1
				}
			}
		}
		modifier = {
			factor = 0.8
			has_estate_privilege = nhs_native_priv_citizenship_for_service
		}
		modifier = {
			factor = 0.5
