country_event = { #Reformation of the Hellenic Faith
	id = nhs_hcd_hellenicflavor.1

	desc = nhs_hcd_hellenicflavor1.d

	is_triggered_only = yes

	trigger = {
		religion = pantheon_worship
		# NOT = {
		# 	has_any_disaster = yes
		# }
	}

	immediate = {
		hidden_effect = {
			set_country_flag = nhs_hellenic_reformed_faith

			clr_country_flag = nhs_hellenic_reformed_poseidon
			clr_country_flag = nhs_hellenic_reformed_apollo
		}
	}

	option = { #Adopt primarily zeus
		name = nhs_hcd_hellenicflavor.1.a

		add_country_modifier = {
			name = nhs_hcf_zeus_worship

		}

		set_country_flag = nhs_hellenic_reformed_zeus
		custom_tooltip = zeus_worship.tt
	}

	option = { #Worship poseidon
		name = nhs_hcd_hellenicflavor.1.b

		add_country_modifier = {
			name = nhs_hcf_poseidon_worship
			duration = -1
		}

		set_country_flag = nhs_hellenic_reformed_poseidon
		custom_tooltip = poseidon_worship.tt
	}

	option = { #Worship apollo
		name = nhs_hcd_hellenicflavor.1.c

		add_country_modifier = {
			name = nhs_hcf_apollo_worship
			duration = -1
		}

		set_country_flag = nhs_hellenic_reformed_apollo
		custom_tooltip = apollo_worship.tt
	}

	option = { #Keep to our traditions
		name = nhs_hcd_hellenicflavor.1.d

		add_country_modifier = {
			name = nhs_hcf_traditional_worship
			duration = -1
		}

		add_stability = 1
		custom_tooltip = traditional_worship.tt
	}

	after = {
		set_country_flag = nhs_hcd_has_reformed_faith	
	}

country_event = { #contact with christian / muslim colonists
	id = nhs_hcd_hellenicflavor.21
	title = nhs_hcd_hellenicflavor21.t
	desc = nhs_hcd_hellenicflavor21.d
	picture = PANTHEON_eventPicture

    trigger = {

		always = no  # -- Disable for now --

		OR = {
            nhs_condition_all_ely = { KOMANDA = tag }
			any_owned_province = {
				is_capital = yes
				continent = north_america
			}
        }
		religion = pantheon_worship
		any_neighbor_country = { 
			OR = {
				religion_group = christian
				religion_group = muslim
				religion_group = jewish_group
			}
			OR = {
				is_colonial_nation = yes
				capital_scope = {
					continent = north_america
				}
			}
			NOT = {
				has_opinion = {
					who = ROOT
					value = 50
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 3
	}

	fire_only_once = yes

	option = { #stand by our ways
		name = nhs_hcd_hellenicflavor.21.a

		add_prestige = 10

		every_known_country = {
			limit = {
				OR = {
					capital_scope = {
						continent = north_america
					}
					colony = 1
				}
				OR = {
					religion_group = christian
					religion_group = pagan
					religion_group = jewish_group
				}
				NOT = {
					OR = {
						alliance_with = ROOT
						has_opinion = {
							who = ROOT
							value = 50
						}
					}
				}
			}
			add_opinion = {
				who = ROOT
				modifier = nhs_hcd_heathen_state
			}
		}
	}

	option = { #make public concessions
		name = nhs_hcd_hellenicflavor.21.b

		add_dip_power = -100

		add_estate_loyalty = {
			estate = estate_church
			loyalty = -10
		}
	}
}

country_event = { #christian reaction to Hellenic threat
	id = nhs_hcd_hellenicflavor.22
	title = nhs_hcd_hellenicflavor22.t
	desc = nhs_hcd_hellenicflavor22.d
	picture = DIPLOMACY_SIGNING_eventPicture

	trigger = {

		always = no  # -- Disable for now --

		nhs_condition_all_ely = { KOMANDA = tag }
		NOT = {
			is_at_war = yes
		}
		religion = pantheon_worship
		any_known_country = {
			religion_group = christian
		}
		num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
			value = 1
			OR = { #check if the hellenic country owns a christian or holy site
			province_id = 379 #jerusalem
			province_id = 118 #roma
			province_id = 151 #constantinople
			province_id = 85 #cologne
			province_id = 206 #santiago
			}
		}

		NOT = {
			has_country_flag = nhs_hcd_religious_coalition
		}
		num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
			value = 25
			OR = {
				continent = europe
				region = maghreb_region
				}
		}
	}

	fire_only_once = yes

	immediate = {
		hidden_effect = {
			set_country_flag = nhs_hcd_religious_coalition
		}
	}

    mean_time_to_happen = {
		months  = 24
	}

	option = { #Christian countries will be given the chance to join a coalition against Elysia
		name = nhs_hcd_hellenicflavor.22.a

		hidden_effect = {
			every_known_country = {
				limit = {
					religion_group = christian
					NOT = {
						is_subject = yes
	
					}
					NOT = { #exlcude high relation countries
						has_opinion = {
							who = ROOT
							value = -60
						}
					}
					NOT = {
						alliance_with = ROOT
					}
					NOT = {
						truce_with = ROOT
					}
					capital_scope = { #Make sure to only target european or neaby christian countries
						OR = {
							continent = europe
							region = maghreb_region
							superregion = near_east_superregion
						}
					}
				}
				country_event = {
					id = nhs_hcd_hellenicflavor.221
				}
			}
		}

		custom_tooltip = nhs_hcd_anti_hellenic_coalition.tt
	}

	option = { #Promise peace
		name = nhs_hcd_hellenicflavor.22.b

		add_war_exhaustion = -2

		add_ruler_modifier = {
			name = nhs_hcd_promised_peace
		}
	}

	option = { #Appease tensions available if enough diplo rep
		name = nhs_hcd_hellenicflavor.22.c

		trigger = {
			diplomatic_reputation = 4
		}

		add_dip_power = -100
	}
}

country_event = { #Coalition against Elysia event (Christian)
	id = nhs_hcd_hellenicflavor.221
	title = nhs_hcd_hellenicflavor221.t
	desc = nhs_hcd_hellenicflavor221.d
	picture = DIPLOMACY_SIGNING_eventPicture

	is_triggered_only = yes

	option = { #Join the coalition
		name = nhs_hcd_hellenicflavor.221.a

		ai_chance = {
			factor = 50
			modifier = { #Cannot join if bankrupt or too unstable
				factor = 0
				OR = {
					war_exhaustion = 10
					is_in_coalition_war = yes
					is_bankrupt = yes
					NOT = { 
						stability = -2
					}
				}
			}
			modifier = { #Encouraged to join if hostile
				factor = 3
				OR = {
					ai_attitude = {
						who = FROM
						attitude = attitude_hostile
					}
					ai_attitude = {
						who = FROM
						attitude = attitude_outraged
					}
				}
			}
			modifier = { #Encourage to join if defender of the faith
				factor = 10
				is_defender_of_faith = yes
			}
		}

		form_coalition_against = FROM
	}

	option = { #Do not join the coalition
		name = nhs_hcd_hellenicflavor.221.b

		add_stability = 1

		ai_chance = {
			factor = 40
			modifier = { #Discouraged if threatened
				factor = 2
				ai_attitude = {
					who = FROM
					attitude = attitude_threatened
				}
			}
			modifier = { #Discouraged if already at war
				factor = 2
				is_at_war = yes
			}
			modifier = { #Discouraged if army is too weak
				factor = 2
				NOT = {
					army_size_percentage = 0.5
				}
			}
		}	
	}
}
