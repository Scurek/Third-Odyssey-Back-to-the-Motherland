	}

	}

			GOD = zeus
		}
	}

	option = {
		name = nhs_hcd_hellenicflavor.100.b

			NOT = { has_personal_deity = nhs_hellenic_zeus }

		}
		to_hellenic_festival_organize = {
			GOD = zeus
		}

			limit = {
				to_hellenic_zeus_pantheon = yes

		}
		else = {
			change_personal_deity = nhs_hellenic_zeus
		}
	}
	
	option = {
		name = nhs_hcd_hellenicflavor.100.c
		highlight = yes

			to_hellenic_festival_can_study = yes
		}
		to_hellenic_festival_study = {
			GOD = zeus
		}
	}
	
	option = {
		name = nhs_hcd_hellenicflavor.1XX.d

	}
