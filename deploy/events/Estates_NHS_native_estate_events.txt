#For the Elysians
country_event = {
	id = natives_estate_events.100
	title = natives_estate_events.100.t
	desc = natives_estate_events.100.d
	picture = NATIVE_CITY_VIEW_eventPicture
	
	#fire_only_once = yes
	
	trigger = {
		nhs_condition_all_ely = { KOMANDA = tag }
		OR = {
			culture_group = elysian_group
			primary_culture = elysian
		}
		NOT = { has_country_flag = native_council_enabled }
		calc_true_if = {
			all_core_province = {
				continent = north_america
				NOT = {
					province_group = mesoamerica_pgroup
				}
				NOT = { region = central_america_region }
				NOT = { has_province_modifier = nhs2_albanian_majority }
				NOT = { has_province_modifier = nhs2_albanian_majority2 }
				owned_by = ROOT
				NOT = { culture = high_elysian }
				is_capital = no
			}
			amount = 5
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		set_country_flag = native_council_enabled
		nhs_barbaroi_check_all = yes
	}
	
	option = {
		name = natives_estate_events.100.a
		custom_tooltip = nhs_enable_nat_estate_ely_tt
		if = {
			limit = {
				has_dlc = "Dharma"
			}
			custom_tooltip = nhs_enable_nat_reform_tt
		}
		if = {
			limit = {
				any_owned_province = {
					continent = north_america
					NOT = {
						province_group = mesoamerica_pgroup
						region = central_america_region
						region = great_plains_region
					}
					NOT = { has_province_modifier = nhs2_albanian_majority }
					NOT = { has_province_modifier = nhs2_albanian_majority2 }
					native_culture = yes
				}
			}
			random_owned_province = {
				limit = {
					continent = north_america
					NOT = {
						province_group = mesoamerica_pgroup
						region = central_america_region
						region = great_plains_region
					}
					NOT = { has_province_modifier = nhs2_albanian_majority }
					NOT = { has_province_modifier = nhs2_albanian_majority2 }
					native_culture = yes
				}
				#set_estate = estate_nhs_native_council
				nhs_add_barbaroi_province = yes
			}
		}
		else_if = {
			limit = {
				continent = north_america
				NOT = {
					province_group = mesoamerica_pgroup
					region = central_america_region
					region = great_plains_region
				}
				NOT = { has_province_modifier = nhs2_albanian_majority }
				NOT = { has_province_modifier = nhs2_albanian_majority2 }
				OR = {
					NOT = { owner = { primary_culture = elysian } }
					OR = {
						any_neighbor_province = {
							native_culture = yes
						}
						any_empty_neighbor_province = {
							native_culture = yes
						}
					}
				}
				NOT = { culture = high_elysian }
			}
			random_owned_province = {
				limit = {
					continent = north_america
					NOT = {
						province_group = mesoamerica_pgroup
						region = central_america_region
						region = great_plains_region
					}
					NOT = { has_province_modifier = nhs2_albanian_majority }
					NOT = { has_province_modifier = nhs2_albanian_majority2 }
					OR = {
						NOT = { owner = { primary_culture = elysian } }
						OR = {
							any_neighbor_province = {
								native_culture = yes
							}
							any_empty_neighbor_province = {
								native_culture = yes
							}
						}
					}
					NOT = { culture = high_elysian }
				}
				#set_estate = estate_nhs_native_council
				nhs_add_barbaroi_province = yes
				if = {
					limit = {
						owner = {
							primary_culture = elysian
						}
					}
					if = {
						limit = {
							any_neighbor_province = {
								native_culture = yes
							}
						}
						random_neighbor_province = {
							limit = {
								native_culture = yes
							}
							PREV = {
								change_culture = PREV
							}
						}
					}
					else_if = {
						limit = {
							any_empty_neighbor_province = {
								native_culture = yes
							}
						}
						random_empty_neighbor_province = {
							limit = {
								native_culture = yes
							}
							PREV = {
								change_culture = PREV
							}
						}
					}
				}
				else_if = {
					limit = {
						owner = {
							culture_group = elysian_group
						}
						NOT = { region = great_plains_region }
					}
					change_culture = elysameric
				}
				else_if = {
					limit = {
						owner = {
							culture_group = norse_g
						}
					}
					change_culture = skraeling
				}
			}
		}
	}
}

country_event = {
	id = natives_estate_events.105
	title = natives_estate_events.100.t
	desc = natives_estate_events.100.d
	picture = NATIVE_CITY_VIEW_eventPicture
	
	#fire_only_once = yes
	hidden = yes
	
	trigger = {
		has_country_flag = native_council_enabled
		any_owned_province = {
			continent = north_america
			NOT = { province_group = mesoamerica_pgroup }
			NOT = { region = central_america_region }
			NOT = { has_province_modifier = nhs2_albanian_majority }
			NOT = { has_province_modifier = nhs2_albanian_majority2 }
			NOT = { culture = high_elysian }
			is_colony = no
			OR = {
				AND = {
					is_state = yes
					NOT = { has_province_flag = nhs_counted_province }
				}
				AND = {
					is_state = no
					has_province_flag = nhs_counted_province
				}
			}
		}
	}
	
	immediate = {
		every_owned_province = {
			limit = {
				continent = north_america
				NOT = { province_group = mesoamerica_pgroup }
				NOT = { region = central_america_region }
				NOT = { has_province_modifier = nhs2_albanian_majority }
				NOT = { has_province_modifier = nhs2_albanian_majority2 }
				NOT = { culture = high_elysian }
				is_colony = no
				OR = {
					AND = {
						is_state = yes
						NOT = { has_province_flag = nhs_counted_province }
					}
					AND = {
						is_state = no
						has_province_flag = nhs_counted_province
					}
				}
			}
			if = {
				limit = {
					is_state = yes
					NOT = { has_province_flag = nhs_counted_province }
				}
				set_province_flag = nhs_counted_province
				ROOT = {
					change_variable = {
						which = nhs_nat_estate_na_provinces_10
						value = 1
					}
					change_variable = {
						which = nhs_nat_estate_na_provinces_20
						value = 1
					}
					if = {
						limit = {
							check_variable = {
								which = nhs_nat_estate_na_provinces_10
								value = 8
							}
						}
						change_variable = {
							which = nhs_nat_estate_balance_10
							value = 1
						}
						change_variable = {
							which = nhs_nat_estate_total_na_provinces_10
							value = 1
						}
						set_variable = {
							which = nhs_nat_estate_na_provinces_10
							value = 0
						}
					}
					if = {
						limit = {
							check_variable = {
								which = nhs_nat_estate_na_provinces_20
								value = 4
							}
						}
						change_variable = {
							which = nhs_nat_estate_balance_20
							value = 1
						}
						change_variable = {
							which = nhs_nat_estate_total_na_provinces_20
							value = 1
						}
						set_variable = {
							which = nhs_nat_estate_na_provinces_20
							value = 0
						}
					}
				}
			}
			else_if = {
				limit = {
					is_state = no
					has_province_flag = nhs_counted_province
				}
				clr_province_flag = nhs_counted_province
				ROOT = {
					change_variable = {
						which = nhs_nat_estate_na_provinces_10
						value = -1
					}
					change_variable = {
						which = nhs_nat_estate_na_provinces_20
						value = -1
					}
					if = {
						limit = {
							NOT = {
								check_variable = {
									which = nhs_nat_estate_na_provinces_10
									value = 0
								}
							}
						}
						change_variable = {
							which = nhs_nat_estate_balance_10
							value = -1
						}
						change_variable = {
							which = nhs_nat_estate_total_na_provinces_10
							value = -1
						}
						set_variable = {
							which = nhs_nat_estate_na_provinces_10
							value = 7
						}
					}
					if = {
						limit = {
							NOT = {
								check_variable = {
									which = nhs_nat_estate_na_provinces_20
									value = 0
								}
							}
						}
						change_variable = {
							which = nhs_nat_estate_balance_20
							value = -1
						}
						change_variable = {
							which = nhs_nat_estate_total_na_provinces_20
							value = -1
						}
						set_variable = {
							which = nhs_nat_estate_na_provinces_20
							value = 3
						}
					}
				}
			}
		}
		calc_barbaroi_angerement = yes
	}
	
	option = {
		name = natives_estate_events.101.a
	}
}

###Events inspired by the Cossacks Estate events
##Colonization of the New World
#Natives vs Nobles - Nobles feel that the natives hold too much influence within the nation
country_event = {
	id = natives_estate_events.0
	title = natives_estate_events.0.t
	desc = natives_estate_events.0.d
	picture = NATIVE_DEBATE_eventPicture
	
	trigger = {
		
		has_estate = estate_nhs_native_council
		has_estate = estate_nobles
		estate_influence = {
			estate = estate_nobles
			influence = 50
		}
		estate_influence = {
			estate = estate_nhs_native_council
			influence = 25
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_nhs_native_council
				modifier = EST_VAL_SERFS_HUNTED
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_nhs_native_council
				modifier = EST_VAL_SERFS_TO_COUNCIL
			}
		}
		any_owned_province = {
			has_estate = estate_nhs_native_council
		}
		any_owned_province = {
			OR = {
				has_estate = estate_nobles
				AND = {
					NOT = { has_estate = estate_nhs_native_council }
					any_neighbor_province = {
						owned_by = ROOT
						has_estate = estate_nhs_native_council
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			set_country_flag = colonization_of_the_steppes_flag
			random_owned_province = {
				limit = {
					has_estate = estate_nhs_native_council
				}
				set_province_flag = serfs_are_arriving
			}
			random_owned_province = {
				limit = {
					OR = {
						has_estate = estate_nobles
						AND = {
							NOT = { has_estate = estate_nhs_native_council }
							any_neighbor_province = {
								owned_by = ROOT
								has_estate = estate_nhs_native_council
							}
						}
					}
					NOT = { has_province_flag = serfs_are_arriving }
				}
				set_province_flag = serfs_are_fleeing
			}
		}
	}
	
	option = {
		name = natives_estate_events.0.a #These Serfs must be brought back.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.5
				NOT = {
					estate_loyalty = {
						estate = estate_nhs_native_council
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.5
				estate_influence = {
					estate = estate_nhs_native_council
					influence = 60
				}
			}
			modifier = {
				factor = 0.5
				estate_influence = {
					estate = estate_nobles
					influence = 60
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_nhs_native_council
			loyalty = -15
		}
		add_estate_influence_modifier = {
			estate = estate_nhs_native_council
			desc = EST_VAL_SERFS_HUNTED
			influence = -10
			duration = 3650
		}
		add_estate_influence_modifier = {
			estate = estate_nobles
			desc = EST_VAL_SERFS_BROUGHT_BACK
			influence = 10
			duration = 3650
		}
		random_owned_province = {
			limit = {
				has_province_flag = serfs_are_fleeing
			}
			add_base_production = 1
			clr_province_flag = serfs_are_fleeing
		}
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_province_flag = serfs_are_arriving
				}
				clr_province_flag = serfs_are_arriving
			}
		}
	}
	option = {
		name = natives_estate_events.0.b #We cannot afford to anger the Native Council.
		ai_chance = {
