#Oracle found!
country_event = {
	id = msg_religionevents.0
	title = msg_religionevents.0.t
	desc = msg_religionevents.0.d
	picture = Oracle_eventPicture

	fire_only_once = yes
	
	trigger = {
		nhs_check_all_spartan = { CONDITION = tag }
		religion_group = pagan
		OR = {
			had_country_flag = { flag = nhs_msg_land1 days = 3650 }
			AND = {
				had_country_flag = { flag = nhs_msg_land1 days = 1825 }
				has_global_flag = nhs_started_at_bookmark
			}
		}
	}
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.1
			has_global_flag = nhs_started_at_bookmark
		}
	}
	
	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = nhs2_oracle_cooldown
				duration = 3650
				hidden = yes
			}
		}
	}
	
	option = {
		name = msg_religionevents.0.a
		add_country_modifier = {
			name = "msg_oracle_before"
			duration = -1
		}
	}
}
#Oracle envisions the rise of the great serpent
country_event = {
	id = msg_religionevents.1
	title = msg_religionevents.1.t
	desc = msg_religionevents.1.d
	picture = Oracle_eventPicture

	fire_only_once = yes
	
	trigger = {
		nhs_check_all_spartan = { CONDITION = tag }
		religion_group = pagan
		has_country_modifier = msg_oracle_before
		has_country_modifier = nhs_high_priests
		OR = {
			had_country_flag = { flag = nhs_msg_land1 days = 7300 }
			AND = {
				had_country_flag = { flag = nhs_msg_land1 days = 5475 }
				has_global_flag = nhs_started_at_bookmark
			}
		}
		NOT = { has_country_flag = aztlan_reformation }
		NOT = { has_global_flag = msg_post_reformation }
	}
	mean_time_to_happen = {
		months = 5 
	}
	option = {
		name = msg_religionevents.1.a
		if = {
			limit = {
				OR = {
					NOT = {	has_country_modifier = nhs_expanded_pantheon }
					NOT = {	has_country_modifier = nhs_consolidated_pantheon }
				}	
			}
			add_country_modifier = {
				name = nhs_consolidated_pantheon
				duration = -1
			}
		}
		every_owned_province = {
			limit = {
				has_owner_religion = no
			}
			add_unrest = 5
		}
	}
	option = {
		name = msg_religionevents.1.b
		if = {
			limit = {
				OR = {
					NOT = {	has_country_modifier = nhs_expanded_pantheon }
					NOT = {	has_country_modifier = nhs_consolidated_pantheon }
				}	
			}
			add_country_modifier = {
				name = nhs_expanded_pantheon
				duration = -1
			}
		}
		every_owned_province = {
			limit = {
				has_owner_religion = yes
			}
			add_unrest = 5
		}
	}
	option = {
		name = msg_religionevents.1.c
		add_prestige = 50
		add_stability = -1
	}
	
}

			owner = { has_country_modifier = nhs_high_priests }
		}

			factor = 0.8
			owner = { has_idea = humanist_tolerance }
		}

			factor = 0.95
			owner = { NOT = { nhs_check_all_elysian_tags_has_opinion_X = { VALUE = -50 } } }

		modifier = {
			factor = 0.85

		modifier = {
			factor = 0.75

		}

			factor = 0.5

			swap_non_generic_missions = yes
		}
	}
