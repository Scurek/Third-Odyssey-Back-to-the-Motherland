# Cultural change for ELysia - Greek to Elysian, native influences
country_event = {
	id = nhs2_cultureevents.100
	title = nhs2_cultureevents.100.t
	desc = nhs2_cultureevents.100.d
	picture = byz_COURT_eventPicture

	fire_only_once = yes
	
	trigger = {
		nhs_condition_all_ely = { KOMANDA = tag }
		has_country_flag = nhs_settled_mission
		is_year = 1500
		primary_culture = greek
		NOT = { has_global_flag = nhs_elysian_culture_change }
		OR = {
			capital_scope = { continent = north_america }
			capital_scope = { continent = south_america }
		}
		OR = {
			technology_group = exiled_greek
			technology_group = modern_greek
		}
		OR = {
			AND = {
				is_year = 1520
				NOT = {
					to_ely_has_integrating_privilege = yes
				}
				calc_true_if = {
					amount = 10
					all_owned_province = {
						to_ely_province_can_integrate = yes
					}
				}
			}
			AND = {
				is_year = 1520
				to_ely_has_integrating_privilege = yes
				calc_true_if = {
					amount = 10
					all_owned_province = {
						to_ely_province_can_integrate = yes
					}
				}
			}
			AND = {
				is_year = 1510
				to_ely_has_integrating_privilege = yes
				calc_true_if = {
					amount = 15
					all_owned_province = {
						to_ely_province_can_integrate = yes
					}
				}
			}
			AND = {
				is_year = 1500
				to_ely_has_integrating_privilege = yes
				calc_true_if = {
					amount = 20
					all_owned_province = {
						to_ely_province_can_integrate = yes
					}
				}
			}
		}
	}

	immediate = {
		set_global_flag = nhs_elysian_culture_change
		set_global_flag = nhs_first_barbaroi_integration
	}

	option = {		# What will be, will be
		name = nhs2_cultureevents.100.a
		change_tag = ELY
		change_primary_culture = elysian
		hidden_effect = {
			if = {
				limit = {
					ruler_culture = greek
				}
				set_ruler_culture = elysian
			}
			if = {
				limit = {
					consort_culture = greek
					has_consort = yes
				}
				set_consort_culture = elysian
			}
			if = {
				limit = {
					heir_culture = greek
					has_heir = yes
				}
				set_heir_culture = elysian
			}
		}
		set_country_flag = nhs_barbaroi_integration
		custom_tooltip = nhs_convert_culture_capital_with_assimilation_tt
		custom_tooltip = nhs_new_line_tt
		nhs_switch_elysian_native_estate_to_mid = yes
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						is_capital = yes
						nhs_homelands = yes
					}
					OR = {
						nhs_culturally_native = yes
						has_province_modifier = to_nat_tribal_development
					}
				}
				if = {
					limit = {
						NOT = { has_albanian_majority_modifier = yes }
					}
					change_culture = elysian
					change_religion = ROOT
				}
				to_nat_assimilate_local_tribal_development = yes
				province_recalculate_estate_privileges_effect_natives = yes
			}
			every_owned_province = {
				limit = {
					culture = greek
					NOT = { continent = europe }
					NOT = { has_albanian_majority_modifier = yes }
				}
				change_culture = elysian
			}
			every_province = {
				limit = {
					OR = {
						continent = north_america
						continent = south_america
					}
					culture = greek
					NOT = {
						owner = {
							primary_culture = greek
						}
					}
				}
				change_culture = elysian
			}
			every_country = {
				country_event = {
					id = nhs2_cultureevents.106
				}
				country_event = {
					id = nhs2_cultureevents.107
				}
			}
		}
	}
}

}

# Gradual cultural conversion of native american to Symmachoi
country_event = {
	id = nhs2_cultureevents.102
	title = nhs2_cultureevents.102.t
	desc = nhs2_cultureevents.102.d
	picture = NATIVES_eventPicture
	
	trigger = {
		culture_group = elysian_group
		has_country_flag = nhs_barbaroi_integration
		any_owned_province = {
			to_ely_can_convert_to_barbaroi = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					to_ely_can_convert_to_barbaroi = yes
				}
				save_event_target_as = nhs_countryevents15_target
			}
		}
	}

	mean_time_to_happen = {
		months = 90
		modifier = {
			factor = 0.8
			has_estate_privilege = nhs_native_priv_citizenship_for_service
		}
		modifier = {
			factor = 0.8
			has_estate_privilege = nhs_native_priv_encourage_barbaroi_officials
		}
		modifier = {
			factor = 0.9
			has_estate_privilege = nhs_native_priv_citizenship_for_service
		}
		modifier = {
			factor = 0.8
			has_active_policy = elysian_integration_act
		}
		modifier = {
			factor = 0.9
			stability = 1
		}
		modifier = {
			factor = 0.9
			stability = 2
		}
		modifier = {
			factor = 0.9
			stability = 3
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 10
				to_ely_can_convert_to_barbaroi = yes
			}
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 15
				to_ely_can_convert_to_barbaroi = yes
			}
		}
		# modifier = {
		# 	factor = 1.5
		# 	NOT = { has_country_modifier = nhs_extended_city_senate }
		# }
	}

	option = {	# Great news
		name = "nhs2_cultureevents.102.a"
		goto = nhs_countryevents15_target
		event_target:nhs_countryevents15_target = {
			change_culture = elysameric
			to_nat_assimilate_local_tribal_development = yes
			nhs_assign_names_ely = yes
			set_province_flag = nhs_culture_changed_to_symmachoi
			hidden_effect = {
				province_recalculate_estate_privileges_effect_natives = yes
			}
		}
	}
}

# Gradual cultural conversion of Symmachoi to Local Elysian group culture
country_event = {
	id = nhs2_cultureevents.103
	title = nhs2_cultureevents.103.t
	desc = nhs2_cultureevents.103.d
	picture = NATIVES_eventPicture
	
	trigger = {
		culture_group = elysian_group
		NOT = { primary_culture = elysameric }
		to_ely_has_integrating_privilege = yes
		any_owned_province = {
			to_ely_can_convert_to_local_elysian_culture = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					to_ely_can_convert_to_local_elysian_culture = yes
				}
				save_event_target_as = nhs_countryevents25_target
				random_neighbor_province = {
					limit = {
						culture_group = elysian_group
						NOT = { culture = elysameric }
						NOT = { culture = shqipproi }
					}
					save_event_target_as = nhs_countryevents25_neighbor_province_target
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 180
		# modifier = {
		# 	factor = 0.5
		# 	has_country_flag = nhs_natives_no_privileges
		# }
		modifier = {
			factor = 0.8
			has_active_policy = elysian_integration_act
		}
		modifier = {
			factor = 0.9
			stability = 1
		}
		modifier = {
			factor = 0.9
			stability = 2
		}
		modifier = {
			factor = 0.9
			stability = 3
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 10
				to_ely_can_convert_to_local_elysian_culture = yes
			}
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 15
				to_ely_can_convert_to_local_elysian_culture = yes
			}
		}
	}

	option = {		# Great news
		name = "nhs2_cultureevents.103.a"
		event_target:nhs_countryevents25_target = {
			if = {
				limit = {
					has_global_flag = nhs_borealians_appeared
					nhs_borealian_province = yes
				}
				change_culture = borealian
			}
			else_if = {
				limit = {
					has_global_flag = nhs_hyperborealians_appeared
					nhs_hyperborealian_province = yes
				}
				change_culture = hyperborealian
			}
			else_if = {
				limit = {
					has_global_flag = nhs_hesperidians_appeared
					nhs_hesperidian_province = yes
				}
				change_culture = hesperidian
			}
			else_if = {
				limit = {
					has_global_flag = nhs_kykladian_appeared
					nhs_kykladian_province = yes
				}
				change_culture = kykladian
			}
			else_if = {
				limit = {
					has_global_flag = nhs_elyric_appeared
					nhs_elyric_province = yes
				}
				change_culture = elyric
			}
			else_if = {
				limit = {
					has_global_flag = nhs_arkadians_appeared
					nhs_arkadian_province = yes
				}
				change_culture = arkadian
			}
			else_if = {
				limit = {
					has_global_flag = nhs_eleutheroi_appeared
					nhs_eleutheroi_province = yes
				}
				change_culture = eleutheroi
			}
			else_if = {
				limit = {
					OR = {
						region = northeast_america_region
						region = southeast_america_region
					}
				}
				change_culture = elysian
			}
			else_if = {
				limit = {
					has_saved_event_target = nhs_countryevents25_neighbor_province_target
					event_target:nhs_countryevents25_neighbor_province_target = {
						culture_group = elysian_group
						NOT = { culture = elysameric }
						NOT = { culture = shqipproi }
					}
				}
				change_culture = event_target:nhs_countryevents25_neighbor_province_target
			}
			else = {
				change_culture = ROOT
			}
			hidden_effect = {
				province_recalculate_estate_privileges_effect_natives = yes
			}
		}
	}
}

# Mesoamerican Integration
country_event = {
	id = nhs2_cultureevents.24
	title = nhs2_cultureevents.24.t
	desc = nhs2_cultureevents.24.d
	picture = GOLDEN_CITY_eventPicture

	goto = nhs_cultural_integration_target
	
	trigger = {
		has_country_flag = nhs_mesoamerican_integration
		any_owned_province = {
			has_province_flag = to_culture_is_being_integrated
			NOT = { has_province_modifier = nhs_mesoamerican_integration_mod }
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_province_flag = to_culture_is_being_integrated
					NOT = { has_province_modifier = nhs_mesoamerican_integration_mod }
				}
				save_event_target_as = nhs_cultural_integration_target
				area = {
					limit = {
						owned_by = ROOT
					}
					clr_province_flag = to_culture_is_being_integrated
				}
			}
		}
	}

	option = {		# Great news
		name = nhs2_cultureevents.24.a
		event_target:nhs_cultural_integration_target = {
			area = {
				limit = {
					owned_by = ROOT
					nhs_mesoamerican_integration_province = yes
					NOT = { has_province_modifier = nhs_mesoamerican_integration_mod }
				}
				if = {
					limit = {
						ROOT = {
							culture_group = elysian_group
						}
					}
					change_culture = lakonian
				}
				else = {
					change_culture = ROOT
				}
			}
		}
	}
}
