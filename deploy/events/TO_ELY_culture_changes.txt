# Cultural changes STAGE ONE - Greek to Elysian (still byzantine group)
country_event = {
	id = nhs2_cultureevents.100
	title = nhs2_cultureevents.100.t
	desc = nhs2_cultureevents.100.d
	picture = byz_COURT_eventPicture
	
	trigger = {
		is_year = 1470
		primary_culture = greek
		OR = {
			capital_scope = { continent = north_america }
			capital_scope = { continent = south_america }
		}
		OR = {
			technology_group = exiled_greek
			technology_group = modern_greek
		}
		NOT = { has_global_flag = nhs_started_at_bookmark }
	}

	option = {		# What will be, will be
		name = "nhs2_cultureevents.100.a"
		change_primary_culture = elysian
		hidden_effect = {
			if = {
				limit = {
					ruler_culture = greek
				}
				set_ruler_culture = elysian
			}
			if = {
				limit = {
					consort_culture = greek
					has_consort = yes
				}
				set_consort_culture = elysian
			}
			if = {
				limit = {
					heir_culture = greek
					has_heir = yes
				}
				set_heir_culture = elysian
			}
		}
		every_owned_province = {
			limit = {
				OR = {
					is_capital = yes
					nhs_homelands = yes
				}
				NOT = { has_province_modifier = nhs2_albanian_majority }
				NOT = { has_province_modifier = nhs2_albanian_majority2 }
				has_province_modifier = to_nat_tribal_development
			}
			change_culture = elysian
			change_religion = ROOT
			to_nat_assimilate_local_tribal_development = yes
		}
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						AND = {
							NOT = { continent = europe }
							culture = greek
						}
						# Free converts (for example: native powhattan if 953 is original)
						is_capital = yes
						nhs_homelands = yes
					}
					NOT = { has_province_modifier = nhs2_albanian_majority }
					NOT = { has_province_modifier = nhs2_albanian_majority2 }
				}
				change_culture = elysian
				nhs_assign_names_ely = yes
			}
		}
	}
}

# Cultural changes STAGE TWO - Elysian to elysian_group with subsets
country_event = {
	id = nhs2_cultureevents.101
	title = nhs2_cultureevents.101.t
	desc = nhs2_cultureevents.101.d
	picture = byz_COURT_eventPicture
	
	trigger = {
		OR = {
			is_year = 1505
			technology_group = modern_greek
		}
		primary_culture = elysian
		OR = {
			has_idea = senate_of_cities
			is_year = 1510
			ai = yes
		}
		NOT = { has_country_modifier = nhs_handicap_settlers }
	}

	immediate = {
		# Change province cultures
		hidden_effect = {
			every_province = {
				limit = {
					culture = elysian
				}
				if = {
					limit = {
						nhs_homelands = yes
						NOT = { has_province_modifier = nhs2_albanian_majority }
						NOT = { has_province_modifier = nhs2_albanian_majority2 }
					}
					change_culture = high_elysian
				}
				else = {
					change_culture = elyric
				}
				nhs_assign_names_ely = yes
			}
			every_subject_country = {
				limit = {
					primary_culture = elysian
				}
				country_event = { id = nhs2_cultureevents.101 }
			}
			if = {
				limit = {
					tag = CM1
				}
				capital_scope = {
					change_culture = cretan_e
				}
				every_owned_province = {
					limit = {
						culture = elysian
					}
					change_culture = cretan_e
				}
				set_country_flag = ctm_culture_union1
			}
			if = {
				limit = {
					tag = MS1
				}
				capital_scope = {
					change_culture = lakonian
				}
				every_owned_province = {
					limit = {
						culture = elysian
					}
					change_culture = lakonian
				}
			}
			else = {
				every_owned_province = {
					limit = {
						native_culture = yes
						NOT = { has_province_modifier = nhs2_albanian_majority }
						NOT = { has_province_modifier = nhs2_albanian_majority2 }
					}
					random = {
						chance = 50
						change_culture = elysameric
						to_nat_assimilate_local_tribal_development = yes
						nhs_assign_names_ely = yes
					}
				}
			}
		}
	}

	option = {		# What will be, will be
		name = "nhs2_cultureevents.101.a"
		if = {
			limit = {
				tag = NHS
			}
			change_tag = ELY
			add_government_reform = senatorial_empire
			change_primary_culture = high_elysian
			if = {
				limit = {
					OR = {
						ruler_culture = elysian
						ruler_culture = greek
					}
				}
				set_ruler_culture = high_elysian
			}
			if = {
				limit = {
					OR = {
						consort_culture = elysian
						consort_culture = greek
					}
					has_consort = yes
				}
				set_consort_culture = high_elysian
			}
			if = {
				limit = {
					OR = {
						heir_culture = elysian
						heir_culture = greek
					}
					has_heir = yes
				}
				set_heir_culture = high_elysian
			}
			if = {
				limit = {
					has_dlc = "Common Sense"
				}
				every_owned_province = {
					limit = {
						nhs_homelands = yes
						OR = {
							culture_group = elysian_group
							culture = elysian
						}
					}
					custom_tooltip = nhs_set_seat_in_paliament_province_tt
					hidden_effect = {
						set_seat_in_parliament = yes
					}
				}
			}
		}
		else_if = {
			limit = {
				NOT = { tag = CM1 }
				NOT = { tag = MS1 }
			}
			change_primary_culture = elyric
			if = {
				limit = {
					ruler_culture = elysian
				}
				set_ruler_culture = elyric
			}
			if = {
				limit = {
					consort_culture = elysian
				}
				set_consort_culture = elyric
			}
			if = {
				limit = {
					heir_culture = elysian
				}
				set_heir_culture = elyric
			}
		}
		else_if = {
			limit = {
				tag = CM1
			}
			change_primary_culture = cretan_e
			if = {
				limit = {
					ruler_culture = elysian
				}
				set_ruler_culture = cretan_e
			}
			if = {
				limit = {
					consort_culture = elysian
				}
				set_consort_culture = cretan_e
			}
			if = {
				limit = {
					heir_culture = elysian
				}
				set_heir_culture = cretan_e
			}
		}
		else_if = {
			limit = {
				tag = MS1
			}
			change_primary_culture = lakonian
			if = {
				limit = {
					ruler_culture = lakonian
				}
				set_ruler_culture = lakonian
			}
			if = {
				limit = {
					consort_culture = lakonian
				}
				set_consort_culture = lakonian
			}
			if = {
				limit = {
					heir_culture = lakonian
				}
				set_heir_culture = lakonian
			}
		}
	}
}

# Gradual cultural conversion of native american to Barbaroi provinces
country_event = {
	id = nhs2_cultureevents.102
	title = nhs2_cultureevents.102.t
	desc = nhs2_cultureevents.102.d
	picture = NATIVES_eventPicture
	
	trigger = {
		culture_group = elysian_group
		any_owned_province = {
			to_ely_can_convert_to_barbaroi = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					to_ely_can_convert_to_barbaroi = yes
				}
				save_event_target_as = nhs_countryevents15_target
			}
		}
	}

	mean_time_to_happen = {
		months = 90
		modifier = {
			factor = 0.8
			has_estate_privilege = nhs_native_priv_citizenship_for_service
		}
		modifier = {
			factor = 0.8
			has_estate_privilege = nhs_native_priv_encourage_barbaroi_officials_early
		}
		modifier = {
			factor = 0.8
			has_active_policy = elysian_integration_act
		}
		modifier = {
			factor = 0.9
			stability = 1
		}
		modifier = {
			factor = 0.9
			stability = 2
		}
		modifier = {
			factor = 0.9
			stability = 3
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 10
				to_ely_can_convert_to_barbaroi = yes
			}
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 15
				to_ely_can_convert_to_barbaroi = yes
			}
		}
		# modifier = {
		# 	factor = 1.5
		# 	NOT = { has_country_modifier = nhs_extended_city_senate }
		# }
	}

	option = {	# Great news
		name = "nhs2_cultureevents.102.a"
		goto = nhs_countryevents15_target
		event_target:nhs_countryevents15_target = {
			change_culture = elysameric
			to_nat_assimilate_local_tribal_development = yes
			nhs_assign_names_ely = yes
		}
	}
}

# Gradual cultural conversion of Elysameric to Elyric provinces
country_event = {
	id = nhs2_cultureevents.103
	title = nhs2_cultureevents.103.t
	desc = nhs2_cultureevents.103.d
	picture = NATIVES_eventPicture
	
	trigger = {
		culture_group = elysian_group
		NOT = { primary_culture = elysameric }
		any_owned_province = {
			to_ely_can_convert_to_elyric = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					to_ely_can_convert_to_elyric = yes
				}
				save_event_target_as = nhs_countryevents25_target
			}
		}
	}

	mean_time_to_happen = {
		months = 300
		# modifier = {
		# 	factor = 0.5
		# 	has_country_flag = nhs_natives_no_privileges
		# }
		modifier = {
			factor = 0.9
			stability = 1
		}
		modifier = {
			factor = 0.8
			has_active_policy = elysian_integration_act
		}
		modifier = {
			factor = 0.9
			stability = 2
		}
		modifier = {
			factor = 0.9
			stability = 3
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 10
				to_ely_can_convert_to_elyric = yes
			}
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 15
				to_ely_can_convert_to_elyric = yes
			}
		}
	}

	option = {		# Great news
		name = "nhs2_cultureevents.103.a"
		event_target:nhs_countryevents25_target = {
			if = {
				limit = {
					AND = {
						nhs_borealian_areas = yes
						has_global_flag = nhs_borealians_appeared
					}
				}
				change_culture = borealian
			}
			else_if = {
				limit = {
					AND = {
						nhs_hyperborealian_areas = yes
						has_global_flag = nhs_hyperborealians_appeared
					}
				}
				change_culture = hyperborealian
			}
			else_if = {
				limit = {
					AND = {
						nhs_hesperidian_areas = yes
						has_global_flag = nhs_hesperidians_appeared
					}
				}
				change_culture = hesperidian
			}
			else_if = {
				limit = {
					AND = {
						region = carribeans_region
						has_global_flag = nhs_kykladian_appeared
					}
				}
				change_culture = kykladian
			}
			else = {
				change_culture = elyric
			}
		}
	}
}

	}

		}
	}

	option = {
		name = "nhs2_cultureevents.7.a"
	}
}

# Borealians appear
country_event = {
	id = nhs2_cultureevents.8
	title = nhs2_cultureevents.8.t
	desc = nhs2_cultureevents.8.d
	picture = niagara_falls_eventPicture
	
	fire_only_once = yes

    trigger = {
		culture_group = elysian_group
		had_country_flag = { flag = nhs_borealians_appear days = 14600 }
		NOT = { has_global_flag = nhs_borealians_appeared }
		OR = {
			nhs_condition_all_ely = { KOMANDA = tag }
			primary_culture = high_elysian
			primary_culture = borealian
		}
		any_owned_province = {
			nhs_borealian_areas = yes
			is_colony = no
			NOT = { culture = shqipproi }
			NOT = { culture = elysameric }
			culture_group = elysian_group
		}
	}
	
	immediate = {
		set_global_flag = nhs_borealians_appeared
		hidden_effect = {
			every_owned_province = {
				limit = {
					is_colony = no
					NOT = { culture = shqipproi }
					NOT = { culture = elysameric }
					OR = {
						nhs_borealian_areas = yes
						AND = {
							NOT = { has_global_flag = nhs_hyperborealians_appeared }
							nhs_hyperborealian_areas = yes
						}	
					}
					NOT = { culture = borealian }
					culture_group = elysian_group
				}
				change_culture = borealian
				if = {
					limit = {
						NOT = { has_province_flag = nhs_got_autonomy }
						owner = { to_gets_autonomy_borealian = yes }
					}
					set_province_flag = nhs_got_autonomy
					add_local_autonomy = 10
				}
			}
		}
	}

	option = {
		name = "nhs2_cultureevents.8.a"
		if = {
			limit = {
				NOT = { has_global_flag = nhs_hyperborealians_appeared }
				any_owned_province = {
					nhs_hyperborealian_areas = yes
				}
			}
			if = {
				limit = {
					to_gets_autonomy_borealian = yes
				}
				custom_tooltip = nhs_spawn_borealian_additional_autonomy_tt
			}
			else = {
				custom_tooltip = nhs_spawn_borealian_additional_tt
			}
		}
		else = {
			if = {
				limit = {
					to_gets_autonomy_borealian = yes
				}
				custom_tooltip = nhs_spawn_borealian_autonomy_tt
			}
			else = {
				custom_tooltip = nhs_spawn_borealian_tt
			}
		}
	}
}

# After flip greek culture fix
country_event = {
	id = nhs2_cultureevents.20
	title = nhs2_cultureevents.17.t
	desc = nhs2_cultureevents.17.d
	picture = niagara_falls_eventPicture
	
	hidden = yes

    trigger = {
		OR = {
			primary_culture = elysian
			culture_group = elysian_group
		}
		any_owned_province = {
			culture = greek
			OR = {
				continent = north_america
				continent = south_america
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					primary_culture = elysian
				}
				every_owned_province = {
					limit = {
						culture = greek
						OR = {
							continent = north_america
							continent = south_america
						}
					}
					change_culture = elysian
				}
			}
			if = {
				limit = {
					culture_group = elysian_group
				}
				every_owned_province = {
					limit = {
						culture = greek
						OR = {
							continent = north_america
							continent = south_america
						}
					}
					change_culture = high_elysian
				}
			}
		}
	}

	option = {
		name = "nhs2_cultureevents.8.a"
	}
}
