#Barbaroi uprising! (ELY)
country_event = {
	id = msg_mission.2
	title = msg_mission.2.t
	desc = msg_mission.2.d
	picture = elysian_army_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		nhs_check_all_elysian_tags = { CONDITION = tag }
	}
	
	immediate = {
		if = {
			limit = {
				NOT = {
					calc_true_if = {
						any_owned_province = {
							has_province_flag = msg_prepped_for_barbaroi_rise_temp
						}
						amount = 5
					}
				}
			}
			every_province = {
				limit = {
					owned_by = ROOT
					has_province_flag = msg_prepped_for_barbaroi_rise_temp
				}
				clr_province_flag = msg_prepped_for_barbaroi_rise_temp
				spawn_rebels = {
					type = particularist_rebels
					size = 3
					culture = elysameric
					unrest = 15
					friend = MSN
				}
				if = {
					limit = {
						NOT = { native_culture = yes }
					}
					change_culture = elysameric
				}
			}
		}
		else_if = {
			limit = {
				NOT = {
					calc_true_if = {
						any_owned_province = {
							has_province_flag = msg_prepped_for_barbaroi_rise_temp
						}
						amount = 10
					}
				}
			}
			every_province = {
				limit = {
					owned_by = ROOT
					has_province_flag = msg_prepped_for_barbaroi_rise_temp
				}
				clr_province_flag = msg_prepped_for_barbaroi_rise_temp
				spawn_rebels = {
					type = particularist_rebels
					size = 2
					culture = elysameric
					unrest = 15
					friend = MSN
				}
				if = {
					limit = {
						NOT = { native_culture = yes }
					}
					change_culture = elysameric
				}
			}
		}
		else = {
			every_province = {
				limit = {
					owned_by = ROOT
					has_province_flag = msg_prepped_for_barbaroi_rise_temp
				}
				clr_province_flag = msg_prepped_for_barbaroi_rise_temp
				spawn_rebels = {
					type = particularist_rebels
					size = 1
					culture = elysameric
					unrest = 15
					friend = MSN
				}
				if = {
					limit = {
						NOT = { native_culture = yes }
					}
					change_culture = elysameric
				}
			}
		}
	}
	
	option = {
		name = msg_mission.2.a
	}
}

#Varangian raiders! (ELY)
country_event = {
	id = msg_mission.4
	title = msg_mission.4.t
	desc = msg_mission.4.d
	picture = norse_army_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		nhs_check_all_elysian_tags = { CONDITION = tag }
	}
	
	immediate = {
		if = {
			limit = {
				VGD = {
					any_owned_province = {
						any_neighbor_province = {
							nhs_check_all_elysian_tags = { CONDITION = owned_by }
						}
					}
				}
			}
			random_owned_province = {
				limit = {
					any_neighbor_province = {
						owned_by = VGD
					}
				}
				spawn_rebels = {
					type = anti_tax_rebels
					size = 5 #note: will never exceed ELY's forcelimit
					culture = varangian_e
					unrest = 5
					win = yes
					friend = MSN
				}
			}
		}
		else = {
			random_owned_province = {
				limit = {
					has_port = yes
					NOT = { province_id = 953 }
				}
				spawn_rebels = {
					type = anti_tax_rebels
					size = 5 #note: will never exceed ELY's forcelimit
					culture = varangian_e
					unrest = 5
					friend = MSN
				}
			}
		}
	}
	
	option = {
		name = msg_mission.4.a
	}
}
