#### Christmas introduced by Orthodox Varangians
country_event = {
	id = eastereggs.1
	title = eastereggs.1.t
	desc = eastereggs.1.d
	picture = yuletide_christmas_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		nhs_condition_all_ely = { KOMANDA = tag }
		OR = {
			religion = orthodox
			religion = elysian_orthodoxy
		}
		any_province = {
			owner = {
				OR = {
					tag = ROOT
					is_subject_of = ROOT
				}
			}
			OR = {
				religion = orthodox
				religion = elysian_orthodoxy
			}
			OR = {
				culture = varangian_e
				culture_group = norse_g
			}
		}
		is_month = 11
		NOT = { is_month = 0 }
	}
	
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.1
			any_owned_province = {
				OR = {
					culture = varangian_e
					culture_group = norse_g
				}
				OR = {
					religion = orthodox
					religion = elysian_orthodoxy
				}
				has_province_modifier = religious_zeal_at_conv
			}
		}
		modifier = {
			factor = 0.5
			any_owned_province = {
				OR = {
					culture = varangian_e
					culture_group = norse_g
				}
				religion = norse_pagan_reformed
			}
		}
		modifier = {
			factor = 0.75
			any_province = {
				owner = {
					is_subject_of = ROOT
				}
				OR = {
					culture = varangian_e
					culture_group = norse_g
				}
				religion = norse_pagan_reformed
			}
		}
	}
	
	# Accept this syncretism (christmas)
	option = {
		name = EASTEREGGNAME1A
		add_prestige = -5
		add_country_modifier = {
			name = yuletide_christmas_syncretism
			duration = 3600
		}
		if = {
			limit = {
				any_owned_province = {
					religion = norse_pagan_reformed
					OR = {
						culture = varangian_e
						culture_group = norse_g
					}
				}
			}
			change_religion = ROOT
			
		}
		else_if = {
			limit = {
				any_province = {
					owner = {
						OR = {
							tag = ROOT
							is_subject_of = ROOT
						}
					}
					OR = {
						religion = orthodox
						religion = elysian_orthodoxy
						religion = norse_pagan_reformed
					}
					OR = {
						culture = varangian_e
						culture_group = norse_g
					}
				}
			}
			change_religion = ROOT
			
		}
	}
	
	# Suppress this syncretism
	option = {
		name = EASTEREGGNAME1B
		if = {
			limit = {
				any_owned_province = {
					OR = {
						religion = orthodox
						religion = elysian_orthodoxy
					}
					OR = {
						culture = varangian_e
						culture_group = norse_g
					}
				}
			}
			change_religion = norse_pagan_reformed
			add_province_modifier = {
				name = religious_zeal_at_conv
				duration = 7200
			}
			random_list = {
				20 = {
					add_unrest = 20
				}
				60 = {
					spawn_rebels = {
						type = religious_rebels
						size = 1
						leader = rebel_preacher
					}
				}
				20 = {
					spawn_rebels = {
						type = religious_rebels
						size = 2
						leader = rebel_preacher
					}
				}
			}
		}
		else_if = {
			limit = {
				any_province = {
					owner = {
						OR = {
							tag = ROOT
							is_subject_of = ROOT
						}
					}
					OR = {
						religion = orthodox
						religion = elysian_orthodoxy
					}
					OR = {
						culture = varangian_e
						culture_group = norse_g
					}
				}
			}
			change_religion = norse_pagan_reformed
			add_province_modifier = {
				name = religious_zeal_at_conv
				duration = 7200
			}
			random_list = {
				20 = {
					add_unrest = 20
				}
				60 = {
					spawn_rebels = {
						type = religious_rebels
						size = 1

					spawn_rebels = {

country_event = {
	id = eastereggs.3
	title = eastereggs.3.t
	desc = eastereggs.3.d
	picture = lady_of_alexandria_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		nhs_is_elysia_late = yes
		OR = {
			has_country_flag = nhs_invasion_second_part
			has_country_flag = nhs_invasion_begins
			has_country_flag = nhs_invasion_over
		}
		owns = 2554
	}
	
	mean_time_to_happen = {
		months = 24
	}
	
	# General
	option = {
		name = eastereggs.3.a
		if = {
			limit = {
				is_female = yes
			}
			add_legitimacy = 5
		}
		define_general = {
			name = "Konstantinia Palaiologina"
			fire = 5
			shock = 4
			manuever = 6
			siege = 3
			female = yes
		}
	}
	
	# Advisor
	option = {
		name = eastereggs.3.b
		if = {
			limit = {
				is_female = yes
			}
			add_legitimacy = 5
		}
		define_advisor = {
			type = konstantinia_palaiologina
			name = "Konstantinia Palaiologina"
			skill = 3
			culture = elysameric
			religion = ROOT
			location = 2554
			discount = yes
			female = yes
		}
	}
	# Sexist
	option = {
		name = eastereggs.3.c
		add_prestige = 5
		if = {
			limit = {
				is_female = yes
			}
			add_legitimacy = -5
		}
		else = {
			add_legitimacy = 5
		}
	}
}
