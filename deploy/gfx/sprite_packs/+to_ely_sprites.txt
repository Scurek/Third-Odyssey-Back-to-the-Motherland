to_elysian_sprite_pack = {
	country_trigger = {
		OR = {
			nhs_check_all_elysian_expanded = { CONDITION = tag }
			tag = SHQ
			tag = MS1
		}
		NOT = { tag = VAE }
	}

	sprite_infantry = {
		0 = ELY_INF_01_S
		1 = ELY_INF_02
		2 = greekgfx_INFANTRY_3
		3 = greekgfx_INFANTRY_4

	sprite_cavalry = {

		OR = {
			nhs_check_all_elysian_expanded = { CONDITION = tag }
			tag = SHQ
			tag = MS1
