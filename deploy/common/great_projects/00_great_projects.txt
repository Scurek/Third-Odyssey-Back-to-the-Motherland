kiel_canal = {
	# province it starts in
	start = 1775
	
	# date built in real life (so anything built during game time will be there if you start a game after that date)
	date = 1895.06.20
	
	#time to build
	time = {
		months = 120
	}
	
	#how much to build one
	build_cost = 10000
	
	#can we move it?
	can_be_moved = no
	
	#tier that the project starts at when first placed in the game (be that at game start or when built by a country as the game progresses)
	starting_tier = 3
	
	#project type
	type = canal

	#can we build it?
	build_trigger = {
		hidden_trigger = {
			NOT = {
				has_great_project = {
					type = kiel_canal_outward_boost
					tier = 3
				}
			}
		}
		FROM = {
			owns_or_vassal_of = 1775
			owns_or_vassal_of = 4141
			is_subject = no			
			OR = { 
				mission_completed = emp_brapru_kiel_canal
				mission_completed = GER_german_industry
				adm_tech = 22
			}
			hidden_trigger = {
				NOT = { mission_completed = to_a_world_empire_mission }
			}
		}
	}
	
	#what to do when it's built
	on_built = {
		add_canal = kiel_canal
		if = {
			limit = {
				owner = {
					is_subject = yes
				}
			}
			FROM = {
				overlord = {
					add_prestige = 30
				}
			}
		}
		else = {
			owner = { 
				add_prestige = 30
			}
		}
		to_nhs_exarch_monument_mod_transfer = yes
	}
	
	#what to do when it's destroyed
	on_destroyed = {
		remove_canal = kiel_canal
	}

	#can our country use it?
	can_use_modifiers_trigger = {

	#can our country upgrade it?
