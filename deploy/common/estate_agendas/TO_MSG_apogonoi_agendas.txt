					NOT = { nhs_culturally_native_primary = yes }

			limit = {
				any_rival_country = {
					NOT = { nhs_culturally_native_primary = yes }

#Restore legitimacy

	can_select = {
		NOT = { legitimacy_equivalent = 65 }
		government = monarchy
		NOT = { has_ruler_modifier = legitimacy_defended }
		NOT = { has_regency = yes }
	}
	selection_weight = {
		factor = 2
		modifier = {
			factor = 1.5
			NOT = { legitimacy_equivalent = 50 }
		}
		modifier = {
			factor = 1.5
			NOT = { legitimacy_equivalent = 35 }
		}
	}
	fail_if = {
		OR = {
			NOT = { government = monarchy }
			has_regency = yes
		}
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_nhs_apogonoi
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_nhs_apogonoi
			}
		}
		else = {
			legitimacy_equivalent = 90
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_nhs_apogonoi
		}
		add_estate_loyalty = {
			estate = estate_nhs_apogonoi
			loyalty = 10
		}

			limit = {
				has_reform = celestial_empire

			add_prestige = 10
		}
		else = {
			add_ruler_modifier = {

			}
		}

    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes

            estate = estate_nhs_apogonoi

			loyalty = -5
			duration = 7300
        }
    }
}

#Annex vassal
estate_apogonoi_annex_subject = {
	can_select = {
		is_at_war = no
		diplomatic_reputation = 0
		any_subject_country = {
			NOT = { total_development = 300 }
			OR = {
				AND = {
					is_vassal = yes
					years_in_vassalage_under = {
						who = root
						years = 10
					}
				}
				AND = {
					is_lesser_in_union = yes
					years_in_union_under = {
						who = root
						years = 50
					}
				}
			}
		}
	}
	selection_weight = {
		factor = 3
	}
	pre_effect = {
		random_subject_country = {
			limit = {
				NOT = { total_development = 300 }
				OR = {
					AND = {
						is_vassal = yes
						years_in_vassalage_under = {
							who = root
							years = 10
						}
					}
					AND = {
						is_lesser_in_union = yes
						years_in_union_under = {
							who = root
							years = 50
						}
					}
				}
			}
			save_event_target_as = agenda_country
			capital_scope = {
				save_event_target_as = agenda_province
			}
			add_country_modifier = {
				name = to_be_agenda_annexed_timer
				duration = 7300
				hidden = yes
			}
		}
	}
	provinces_to_highlight = {
		OR = {
			province_id = event_target:agenda_province
			owned_by = event_target:agenda_country
		}
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_nhs_apogonoi
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_nhs_apogonoi
			}
		}
		else = {
			event_target:agenda_country = {
				NOT = { num_of_cities = 1 }
			}
			event_target:agenda_province = {
				owned_by = root
			}
		}
	}
	fail_if = {
		OR = {
			AND = {
				event_target:agenda_country = {
					NOT = { num_of_cities = 1 }
				}
				NOT = {
					event_target:agenda_province = {
						owned_by = root
					}
				}
			}
			event_target:agenda_country = {
				num_of_cities = 1
				NOT = { is_subject_of = root }
			}
		}
	}
	task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_nhs_apogonoi
		}
		add_estate_loyalty = {
			estate = estate_nhs_apogonoi
			loyalty = 10
		}
		custom_tooltip = no_penalties_annexing_tt
	}
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_nhs_apogonoi
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
		hidden_effect = {
			event_target:agenda_country = {
				remove_country_modifier = to_be_agenda_annexed_timer
			}
		}
    }
}
