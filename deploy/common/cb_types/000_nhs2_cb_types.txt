cb_nhs_invasion = {
	valid_for_subject = no

	prerequisites = {
		OR = {
			capital_scope = { continent = north_america }
			capital_scope = { continent = south_america }
		}
		has_country_flag = nhs_invasion_second_part_begins
		NOT = { has_country_flag = nhs_invasion_over }
		is_free_or_tributary_trigger = yes
		FROM = {
			NOT = {
				AND = {
					culture_group = byzantine
					NOT = { primary_culture = turkish }
				}
			}
			any_owned_province = {
				continent = europe
				culture = greek
			}			
		}
	}
	
	war_goal = nhs_invasion
}

cb_north_american_hegemony = {
	valid_for_subject = no

	prerequisites = {
		nhs_condition_all_ely = { KOMANDA = tag }
		is_free_or_tributary_trigger = yes
		has_country_flag = nhs_completed_inward_focus
		
		NOT = { has_country_flag = nhs_completed_promised_land }
		NOT = { nhs_is_revolution_target = yes }
		FROM = {
			NOT = { is_subject_of = ROOT }
			NOT = { alliance_with = ROOT }
			OR = {
				has_country_flag = nhs_expanding_south_completey
				nhs_check_all_vinland = { CONDITION = tag }
				AND = {
					ROOT = {
						has_country_flag = nhs_expanding_north_completed
					}
					OR = {
						nhs_check_all_spartan = { CONDITION = tag }
						capital_scope = {
							province_group = mesoamerica_pgroup
						}
					}
				}
			}
			any_owned_province = {
				continent = north_america
			}
		}
	}
	
	war_goal = nhs_wargoal_north_american_hegemony
}
