cb_american_hegemony = {
	valid_for_subject = no

	prerequisites = {
		mission_completed = nhs_inward_focus
		FROM = {
			NOT = { is_subject_of = ROOT }
			if = {
				limit = {
					ROOT = {
						mission_completed = nhs_the_promised_land
					}
				}
				any_owned_province = {
					OR = {
						continent = north_america
						continent = south_america
					}
				}
			}
			else_if = {
				limit = {
					ROOT = {
						mission_completed = nhs_expanding_south
					}
				}
				any_owned_province = {
					continent = north_america
				}
			}
			else_if = {
				limit = {
					ROOT = {
						mission_completed = nhs_expanding_north
					}
				}
				OR = {
					nhs_check_all_vinland = { CONDITION = tag }
					nhs_check_all_spartan = { CONDITION = tag }
					capital_scope = {
						province_group = mesoamerica_pgroup
					}
				}
				any_owned_province = {
					continent = north_america
				}
			}
			else = {
				nhs_check_all_vinland = { CONDITION = tag }
				any_owned_province = {
					continent = north_america
				}
			}
		}
	}
	
	war_goal = nhs_wargoal_american_hegemony
}
