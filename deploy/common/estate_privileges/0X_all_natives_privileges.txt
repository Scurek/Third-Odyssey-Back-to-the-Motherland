nhs_native_priv_expansionist_coexistance_early = {
    icon = privilege_grant_new_world_charter

	land_share = 10
	loyalty = 0.05
	influence = 0.15

	is_valid = {
		has_country_flag = native_council_enabled_early
	}
	on_invalid = {
		if = {
			limit = {
				has_country_flag = native_council_enabled_mid
			}
			set_estate_privilege = nhs_native_priv_expansionist_coexistance_mid
		}
	}
	penalties = {
	}
	benefits = {
	}
	ai_will_do = {
		factor = 5
	}

	modifier_by_land_ownership = {
		global_colonial_growth = 100
	}
}

# Local councils recognized

bns_native_priv_local_councils_recognized_early = {
    icon = privilege_royal_authority

	loyalty = 0.05
	influence = 0.10

	is_valid = {
		has_country_flag = native_council_enabled_early
	}
	on_granted = {
		custom_tooltip = bns_native_priv_local_councils_recognized_early_effect_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					nhs_culturally_native = yes
				}
				add_province_modifier = {
					name = bns_native_priv_local_councils_recognized_early_mod
					duration = -1
				}
			}
		}
	}
	on_revoked = {
		custom_tooltip = bns_native_priv_local_councils_recognized_early_revoke_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = bns_native_priv_local_councils_recognized_early_mod
				}
				remove_province_modifier = bns_native_priv_local_councils_recognized_early_mod
			}
		}
	}
	on_invalid = {
		custom_tooltip = bns_native_priv_local_councils_recognized_early_revoke_tooltip
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = bns_native_priv_local_councils_recognized_early_mod
				}
				remove_province_modifier = bns_native_priv_local_councils_recognized_early_mod
			}
		}
		if = {
			limit = {
				has_country_flag = native_council_enabled_mid
			}
			set_estate_privilege = bns_native_priv_local_councils_recognized_mid
		}
	}
	penalties = {
	}
	benefits = {
	}
	ai_will_do = {
		factor = 5
	}
}
