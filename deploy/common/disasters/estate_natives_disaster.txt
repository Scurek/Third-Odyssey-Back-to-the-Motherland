estate_all_natives_disaster = {
	# Natives in country rise up, oppose aggression against other natives

	potential = {
		nhs_has_any_native_estate = yes
	}

	can_start = {
		if = {
			limit = {
				ai = yes
			}
			always = no
		}
		nhs_has_any_native_estate = yes
		# has_any_disaster = no
		any_country = {
			defensive_war_with = ROOT
			has_native_government = yes
			nhs_culturally_native_primary = yes
		}
		any_owned_province = {
			is_colony = no
			to_nat_tribal_native_culture_province = yes
		}
		nhs_check_influence_any_native_estate = { QUANTITY = 30 }
	}
	
	can_stop = {
		OR = {
			# has_any_disaster = yes
			NOT = {
				any_country = {
					defensive_war_with = ROOT
					has_native_government = yes
					nhs_culturally_native_primary = yes
				}
			}
			NOT = {
				nhs_check_influence_any_native_estate = { QUANTITY = 30 } 
			}
			NOT = {
				any_owned_province = {
					is_colony = no
					to_nat_tribal_native_culture_province = yes
				}
			}
		}
	}

	down_progress = {
		# Note: apparently, disaster can only tick down when stop conditons apply (?)
		factor = 0
		modifier = {
			factor = 0.1
			is_at_war = no
		}
		modifier = {
			factor = 0.25
			is_at_war = no
			nhs_check_loyalty_any_native_estate = { QUANTITY = 60 }
		}
		modifier = {
			factor = 0.25
			is_at_war = no
			nhs_check_loyalty_any_native_estate = { QUANTITY = 70 }
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			if = {
				limit = {
					ai = yes
				}
				always = no
			}
			any_country = {
				defensive_war_with = ROOT
				has_native_government = yes
				nhs_culturally_native_primary = yes
			}
		}
		modifier = {
			factor = 0.5
			if = {
				limit = {
					ai = yes
				}
				always = no
			}
			any_country = {
				defensive_war_with = ROOT
				has_native_government = yes
				nhs_culturally_native_primary = yes
			}
			NOT = {
				nhs_check_loyalty_any_native_estate = { QUANTITY = 50 }
			}
		}
		modifier = {
			factor = 0.5
			if = {
				limit = {
					ai = yes
				}
				always = no
			}
			any_country = {
				defensive_war_with = ROOT
				has_native_government = yes
				nhs_culturally_native_primary = yes
			}
			estate_influence = {
				estate = estate_all_natives
				influence = 60
			}
		}
	}
	
	modifier = {
		# Own army worse
		land_morale = -0.20
		manpower_recovery_speed = -0.5
		# Mercenaries difficult to find
		mercenary_cost = 1.0
		mercenary_manpower = -0.5
	}
	
	can_end = {
		NOT = {
			any_country = {
				defensive_war_with = ROOT
				has_native_government = yes
				nhs_culturally_native_primary = yes
			}
		}
		NOT = { num_of_rebel_controlled_provinces = 1 }
		NOT = { num_of_rebel_armies = 1 }
		custom_trigger_tooltip = {
			tooltip = nhs_natives_disaster_timer_tt
			NOT = { has_country_modifier = nhs_natives_disaster_timer }
		}
	}

	on_start = nhs2_estateevents.1
	on_end = nhs2_estateevents.2
	
	on_monthly = {
		events = {
			# TODO: events after 1 and 2 years for earlier negotiations
		}
		random_events = { 
			1000 = 0
			# TODO: raise more rebels, neighboring native negative opinion, native subject increase liberty desire, 
			#  colony suffers setback (lose colony progress, add temporary country modifier),
			#  defections in military (tradition, manpower loss), empower enemies (move development to hostile native)
			# 100 = nhs2_estateevents.100
		}
	}
}
