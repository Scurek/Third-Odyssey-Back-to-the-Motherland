estate_all_natives_disaster = {
	potential = {
		has_dlc = "The Cossacks"
		nhs_has_any_native_estate = yes
		estate_influence = {
			estate = estate_all_natives
			influence = 80
		}
		OR = {
			NOT = { has_country_flag = cossack_native_council_triggered }
			had_country_flag = {
				flag = cossack_native_council_triggered
				days = 365
			}
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_all_natives
			influence = 100
		}
	}
	
	can_stop = {
		NOT = {
			estate_influence = {
				estate = estate_all_natives
				influence = 100
			}	
		}
	}
	
	progress = {
		modifier = {
			factor = 5
			estate_influence = {
				estate = estate_all_natives
				influence = 100
			}
		}
		modifier = {
			factor = 1
			NOT = {
				nhs_check_loyalty_any_native_estate = { QUANTITY = 35 }
			}
		}
	}
	
	can_end = {
		always = yes
	}

	on_start = nhs2_estateevents.1
	
	on_monthly = {
	}
}
