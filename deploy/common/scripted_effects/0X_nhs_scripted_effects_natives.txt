# Scripted effects tied to the Third Odyssey natives estate


province_recalculate_estate_privileges_effect_natives = {
    if = {
        # Owner has the estate (culture changed, or owner swapped to nation with estate),
        #  and province has natives culture:
        #  recalculate modifier presences.
        limit = {
            owner = {has_estate = estate_all_natives}
            OR = {
                native_culture = yes
                mesoamerican_native_culture = yes
                southamerican_native_culture = yes
            }
        }

        # Early modifiers

        if = {
            limit = {
                NOT = { has_province_modifier = bns_native_priv_local_councils_recognized_early_mod }
                owner = { has_estate_privilege = bns_native_priv_local_councils_recognized_early }
            }
            add_province_modifier = {
                name = bns_native_priv_local_councils_recognized_early_mod
                duration = -1
            }
        }
        else_if = {
            limit = {
                has_province_modifier = bns_native_priv_local_councils_recognized_early_mod
                owner = { NOT = { has_estate_privilege = bns_native_priv_local_councils_recognized_early } }
            }
            remove_province_modifier = bns_native_priv_local_councils_recognized_early_mod
        }

        # Mid modifiers


        # Late modifiers


    }
    else = {
        # Owner does not have the estate (owner changed, or estate somehow invalid),
        #  or province does not have native culture:
        #  remove all modifiers.

        # Early modifiers
        if = {
            limit = {
                has_province_modifier = bns_native_priv_local_councils_recognized_early_mod
            }
            remove_province_modifier = bns_native_priv_local_councils_recognized_early_mod
        }

        # Mid modifiers


        # Late modifiers


    }
	

}
