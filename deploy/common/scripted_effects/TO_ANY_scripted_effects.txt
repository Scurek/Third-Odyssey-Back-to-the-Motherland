to_compute_non_colonial_development = {
	set_variable = {
		which = $VARIABLE$
		value = 0
	}
	every_owned_province = {
		limit = {
			is_city = yes
		}
		PREV = {
			export_to_variable = {
				which = to_local_dev
				value = trigger_value:development
				who = PREV
			}
			change_variable = {
				which = $VARIABLE$
				which = to_local_dev
			}
		}
	}
}

to_compute_current_year_in_byzantine_calendar = {
	export_to_variable = {
		which = to_currentYearByzantineCalendar
		value = trigger_value:is_year
	}
	change_variable = {
		which = to_currentYearByzantineCalendar
		value = 5508
	}
	if = {
		limit = {
			is_month = 8
		}
		change_variable = {
			which = to_currentYearByzantineCalendar
			value = 1
		}
	}
}

to_setup_non_hellenic_pagan_provinces = {
	every_province = {
		limit = {
			religion_group = pagan
			NOT = { religion = pantheon_worship }
		}
		set_province_flag = to_non_hellenic_pagan_religion
	}
}

to_check_non_hellenic_pagan_provinces_manual_conversion = {
	if = {
		limit = {
			has_province_flag = to_converted_heretics_as_hellenic_potential
		}
		clr_province_flag = to_converted_heretics_as_hellenic_potential
		owner = {
			set_country_flag = to_converted_heretics_as_hellenic
		}
	}
}

to_check_non_hellenic_pagan_provinces = {
	clr_province_flag = to_converted_heretics_as_hellenic_potential
	if = {
		limit = {
			has_province_flag = to_non_hellenic_pagan_religion
			OR = {
				religion = pantheon_worship
				NOT = { religion_group = pagan }
			}
		}
		clr_province_flag = to_non_hellenic_pagan_religion
		if = {
			limit = {
				religion = pantheon_worship
				owner = {
					religion = pantheon_worship
				}
			}
			set_province_flag = to_converted_heretics_as_hellenic_potential
		}
	}
	else_if = {
		limit = {
			NOT = { has_province_flag = to_non_hellenic_pagan_religion }
			religion_group = pagan
			NOT = { religion = pantheon_worship }
		}
		set_province_flag = to_non_hellenic_pagan_religion
	}
}

to_save_years_of_nationalism = {
	hidden_effect = {
		set_variable = {
			which = to_saved_years_of_nationalism
			value = 0
		}
		if = {
			limit = {
				nationalism = 1
				NOT = { has_province_flag = to_separatism_restore_pending }
			}
			export_to_variable = {
				which = to_saved_years_of_nationalism
				value = trigger_value:nationalism
			}
			subtract_variable = {
				which = to_saved_years_of_nationalism
				value = 2
			}
			if = {
				limit = {
					has_province_modifier = to_kastrioti_dominion_penalty
				}
				change_variable = {
					which = to_saved_years_of_nationalism
					value = -10
				}
			}
		}
	}
}

					AND = {

								mission_completed = to_the_old_gods_mission

to_compute_merc_company_size = {
	export_to_variable = {
		which = to_total_development
		value = total_development
	}
	set_variable = {
		which = $OUTPUT_VARIABLE$
		which = to_total_development
	}
	multiply_variable = {
		which = $OUTPUT_VARIABLE$
		value = $REGIMENTS_PER_DEVELOPMENT$
	}
	to_round_down_variable = { VARIABLE = $OUTPUT_VARIABLE$ }
	if = {
		limit = {
			mil_tech = 16
		}
		set_variable = {
			which = to_merc_size_cap
			value = 30
		}
	}
	else_if = {
		limit = {
			mil_tech = 14
		}
		set_variable = {
			which = to_merc_size_cap
			value = 29
		}
	}
	else_if = {
		limit = {
			mil_tech = 11
		}
		set_variable = {
			which = to_merc_size_cap
			value = 27
		}
	}
	else_if = {
		limit = {
			mil_tech = 9
		}
		set_variable = {
			which = to_merc_size_cap
			value = 25
		}
	}
	else_if = {
		limit = {
			mil_tech = 6
		}
		set_variable = {
			which = to_merc_size_cap
			value = 24
		}
	}
	else_if = {
		limit = {
			mil_tech = 5
		}
		set_variable = {
			which = to_merc_size_cap
			value = 22
		}
	}
	else_if = {
		limit = {
			mil_tech = 2
		}
		set_variable = {
			which = to_merc_size_cap
			value = 20
		}
	}
	else = {
		set_variable = {
			which = to_merc_size_cap
			value = 15
		}
	}
	set_variable = {
		which = to_relative_size_mod
		value = $REGIMENTS_PER_DEVELOPMENT$
	}
	divide_variable = {
		which = to_relative_size_mod
		value = 0.05
	}
	multiply_variable = {
		which = to_merc_size_cap
		which = to_relative_size_mod
	}
	to_round_down_variable = { VARIABLE = to_merc_size_cap }
	if = {
		limit = {
			check_variable = {
				which = $OUTPUT_VARIABLE$
				which = to_merc_size_cap
			}
		}
		set_variable = {
			which = $OUTPUT_VARIABLE$
			which = to_merc_size_cap
		}
	}
	set_variable = {
		which = to_total_development
		which = 0
	}
	set_variable = {
		which = to_merc_size_cap
		which = 0
	}
	set_variable = {
		which = to_relative_size_mod
		which = 0
	}
}

to_change_manpower_by_VARIABLE = {

		which = to_manpower_change_counter
		which = $VARIABLE$
	}

				value = 10000
			}
		}

			which = to_manpower_change_counter
			value = 10000
		}

			add_manpower = $SIGN$10
		}
	}

			}
		}
		subtract_variable = {
			which = to_manpower_change_counter
			value = 1000
		}
		$SCOPE$ = {
			add_manpower = $SIGN$1
		}
	}

			value = 100

		}

		limit = {
			check_variable = {
				which = to_manpower_change_counter
				value = 10
			}
		}
		subtract_variable = {
			which = to_manpower_change_counter
			value = 10
		}
		$SCOPE$ = {
			add_manpower = $SIGN$0.01
		}
	}

	}

to_add_ruler_heir_personality_or_modifier = {
	if = {
		limit = {
			has_dlc = "Rights of Man"
			OR = {
				has_regency = no
				has_consort_regency = yes
			}
			OR = {
				NOT = { num_of_ruler_traits = { amount = 3 } }
				ruler_has_bad_adm_personality = yes
				ruler_has_bad_dip_personality = yes
				ruler_has_bad_mil_personality = yes
			}
		}
		if = {
			limit = {
				num_of_ruler_traits = { amount = 3 }
			}
			if = {
				limit = {
					ruler_has_bad_adm_personality = yes
				}
				to_remove_bad_personality_adm = { WHO = ruler }
			}
			else_if = {
				limit = {
					ruler_has_bad_dip_personality = yes
				}
				to_remove_bad_personality_dip = { WHO = ruler }
			}
			else_if = {
				limit = {
					ruler_has_bad_mil_personality = yes
				}
				to_remove_bad_personality_mil = { WHO = ruler }
			}
		}
		# Ensure the added trait is displayed fist
		hidden_effect = {
			to_remove_and_save_personalities = { WHO = ruler }
		}
		add_ruler_personality = $RULER_PERSONALITY$
		hidden_effect = {
			to_restore_personalities = { WHO = ruler }
		}
	}
	else_if = {
		limit = {
			OR = {
				has_regency = no
				has_consort_regency = yes
			}
		}
		add_ruler_modifier = {
			name = $RULER_MODIFIER$
			duration = -1
		}
	}

	if = {
		limit = {
			has_dlc = "Rights of Man"
			has_heir = yes
			has_regency = yes
		}
		if = {
			limit = {
				OR = {
					heir_has_good_adm_personality = yes
					heir_has_good_dip_personality = yes
					heir_has_good_mil_personality = yes
					heir_has_bad_adm_personality = yes
					heir_has_bad_dip_personality = yes
					heir_has_bad_mil_personality = yes
				}
			}
			set_heir_flag = $HEIR_FLAG$
			custom_tooltip = $HEIR_TOOLTIP_PERSONALITY$
		}
		else = {
			add_heir_personality = $HEIR_PERSONALITY$
		}
	}
	else_if = {
		limit = {
			has_heir = yes
			has_regency = yes
		}
		set_heir_flag = $HEIR_FLAG$
		custom_tooltip = $HEIR_TOOLTIP_MODIFIER$
	}
}

			save_event_target_as = to_overlord_overlord
		}
		every_subject_country = {

				is_colonial_nation = yes
			}

			set_country_flag = to_former_overlord@FROM
			every_owned_province = {
				set_province_flag = to_former_colonial_owner@FROM
			}
			to_transfer_colonial_nation = {
				OVERLORD_SCOPE = event_target:to_overlord_overlord
				COLONY_SCOPE = PREV
			}
			country_event = { id = nhs2_test.6 }
		}
	}
}

to_handle_subject_colonial_nations_on_losing_dependancy = {
	if = {
		limit = {
			exists = yes
			ROOT = {
				colony = 1
			}
		}
		ROOT = {
			every_subject_country = {
				limit = {
					is_colonial_nation = yes
					has_country_flag = to_former_overlord@FROM
				}
				clr_country_flag = to_former_overlord@FROM
				to_transfer_colonial_nation = {
					OVERLORD_SCOPE = FROM
					COLONY_SCOPE = PREV
				}
				every_owned_province = {
					if = {
						limit = {
							NOT = { has_province_flag = to_former_colonial_owner@FROM }
						}
						if = {
							limit = {
								is_state_core = PREV
							}
							add_core = ROOT
						}
						else_if = {
							limit = {
								is_territorial_core = PREV
							}
							add_territorial_core = ROOT
						}
						cede_province = ROOT
					}
					clr_province_flag = to_former_colonial_owner@FROM
				}
			}
		}
	}
}
