nhs_effect_for_existing_elysian_tag = {
	if = {
		limit = {
			exists = NHS
		}
		NHS = {
			$EFFECT$
		}
	}
	else_if = {
		limit = {
			exists = ELY
		}
		ELY = {
			$EFFECT$
		}
	}
	else_if = {
		limit = {
			exists = ELR
		}
		ELR = {
			$EFFECT$
		}
	}
	else_if = {
		limit = {
			exists = ERR
		}
		ERR = {
			$EFFECT$
		}
	}
	else_if = {
		limit = {
			exists = ER1
		}
		ER1 = {
			$EFFECT$
		}
	}
}
