to_count_provinces_in_region = {
	set_variable = {
		which = $REGION$_provinces
		value = 0
	}
	every_province = {
		limit = {
			region = $REGION$
		}
		ROOT = {
			change_variable = {
				which = $REGION$_provinces
				value = 1
			}
		}
	}
}
