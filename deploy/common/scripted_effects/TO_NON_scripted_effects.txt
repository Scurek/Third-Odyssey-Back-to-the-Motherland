to_ongoing_fixes = {
	if = {
		limit = {
			950 = {
				has_province_modifier = nhs_seagate_guardians_tm
			}
		}
		950 = {
			show_ambient_object = wonder_statue
			show_ambient_object = wonder_statue_2
		}
	}
	if = {
		limit = {
			has_native_government = yes
			has_regency = yes
			OR = {
				has_ruler = "Native Council"
				has_ruler = "Native Council (Interregnum)"
			}
		}
		define_ruler = {
			culture = ROOT
			religion = ROOT
		}
	}
	if = {
		limit = {
			exists = MS1
			exists = MSN
		}
		MSN = {
			inherit = MS1
		}
	}

	if = {
		limit = {
			has_country_modifier = nhs2_invasion_succeseful
			has_country_flag = to_integrated_greeks
		}
		remove_country_modifier = nhs2_invasion_succeseful
		add_country_modifier = {
			name = nhs2_invasion_succeseful_2
			duration = -1
		}
	}
}

to_count_provinces_in_region = {
	set_variable = {
		which = $REGION$_provinces
		value = 0
	}
	every_province = {
		limit = {
			region = $REGION$
		}
		ROOT = {
			change_variable = {
				which = $REGION$_provinces
				value = 1
			}
		}
	}
}
