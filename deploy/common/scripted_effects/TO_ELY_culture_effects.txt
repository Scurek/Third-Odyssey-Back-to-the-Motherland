to_roman_culture_event_immediate = {
	hidden_effect = {
		set_country_flag = nhs_had_$CULTURE$_spawn_event
		clr_country_flag = nhs_$CULTURE$_event_bounce
		if = {
			limit = {
				NOT = { to_has_culture_appeared = { CULTURE = $CULTURE$ } }
			}
			set_global_flag = nhs_$CULTURE$_event_block
		}
	}
}

to_set_culture_appeared_flag = {

to_spawn_roman_subculture = {
	hidden_effect = {
		to_set_culture_appeared_flag = {
			CULTURE = $CULTURE$
		}
		clr_global_flag = nhs_$CULTURE$_event_block
	}
	if = {
		limit = {
			NOT = { nhs_check_all_elysian_tags = { CONDITION = tag } }
			capital_scope = {
				$PROVINCE_TRIGGER$
			}
		}
		hidden_effect = {
			every_province = {
				limit = {
					owned_by = ROOT
					NOT = { culture = ROOT }
					has_construction = culture
					NOT = { has_albanian_majority_modifier = yes }
				}
				set_province_flag = to_check_culture_after_conversion
			}
		}
		to_primary_culture_and_ruler_heir_consort_culture_if_primary = {
			CULTURE = $CULTURE$
			PRIMARY_CULTURE = THIS
		}
	}
	# Replaced with manual parameter for more flexibility.
	# custom_tooltip = nhs_spawn_$CULTURE$_tt
	$CULTURE_SPAWN_TOOLTIP$
	hidden_effect = {
		every_province = {
			limit = {
				owner = {
					OR = {
						tag = ROOT
						NOT = { nhs_post_fragmentation_culturally_elysian = yes }
					}
				}
				OR = {
					to_nhs_has_dynamic_elysian_culture = yes
					[[ADDITIONAL_CULTURE_TRIGGER] $ADDITIONAL_CULTURE_TRIGGER$]
				}
				NOT = { culture = $CULTURE$ }
				$PROVINCE_TRIGGER$
			}
			change_culture = $CULTURE$
		}
		every_province = {
			limit = {
				owned_by = ROOT
				has_construction = culture
				$PROVINCE_TRIGGER$
			}
			set_province_flag = to_check_culture_after_conversion
		}
	}
	if = {
		limit = {
			any_country = {
				nhs_post_fragmentation_culturally_elysian = yes
				NOT = { has_country_flag = nhs_had_$CULTURE$_spawn_event }
				OR = {
					overlord_of = ROOT
					any_subject_country = {
						overlord_of = ROOT
					}
					any_owned_province = {
						$PROVINCE_TRIGGER$
						OR = {
							to_nhs_has_dynamic_elysian_culture = yes
							[[ADDITIONAL_CULTURE_TRIGGER] $ADDITIONAL_CULTURE_TRIGGER$]
						}
						is_colony = no
					}
				}
			}
		}
		custom_tooltip = nhs_new_line_tt
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { tag = PREV }
					nhs_post_fragmentation_culturally_elysian = yes
					NOT = { has_country_flag = nhs_had_$CULTURE$_spawn_event }
					OR = {
						overlord_of = ROOT
						any_subject_country = {
							overlord_of = ROOT
						}
						any_owned_province = {
							$PROVINCE_TRIGGER$
							OR = {
								to_nhs_has_dynamic_elysian_culture = yes
								[[ADDITIONAL_CULTURE_TRIGGER] $ADDITIONAL_CULTURE_TRIGGER$]
							}
							is_colony = no
						}
					}
				}
				set_country_flag = nhs_$CULTURE$_event_bounce
				country_event = { id = $EVENT$ }
			}
		}
		tooltip = {
			every_country = {
				limit = {
					NOT = { tag = PREV }
					nhs_post_fragmentation_culturally_elysian = yes
					NOT = { has_country_flag = nhs_had_$CULTURE$_spawn_event }
					OR = {

								[[ADDITIONAL_CULTURE_TRIGGER] $ADDITIONAL_CULTURE_TRIGGER$]

to_nhs_change_culture_to_romaic_or_elysian = {
	if = {
		limit = {
			to_nhs_eligible_for_romaic_culture = yes
		}
		if = {
			limit = {
				NOT = { culture = romaic }
			}
			change_culture = romaic
		}
	}
	else_if = {
		limit = {
			NOT = { culture = elysian }
		}
		change_culture = elysian
	}
}

to_nhs_change_original_culture_to_romaic_or_elysian = {
	if = {
		limit = {
			to_nhs_eligible_for_romaic_culture = yes
		}
		change_original_culture = romaic
	}
	else = {
		change_original_culture = elysian
	}
}
