to_check_exarch_province = {
	if = {
		limit = {
			owner = {
				nhs_check_all_elysian_tags = { CONDITION = tag }
			}
			to_potential_exarch_province = yes
			to_exarch_can_get_province_on_owner_change = yes
		}
		owner = {
			save_event_target_as = to_exarch_province_recipient
		}
		to_exarch_effect_from_province = {
			EFFECT = "
			if = {
				limit = {
					exists = yes
					is_subject_of = event_target:to_exarch_province_recipient
					is_subject_of_type = elysian_subject
				}
				save_event_target_as = to_exarch_province_recipient
			}
			"
		}
		if = {
			limit = {
				owner = {
					has_saved_event_target = to_exarch_province_recipient
					event_target:to_exarch_province_recipient = {
						is_subject_of = PREV
						is_subject_of_type = elysian_subject
					}	
				}
			}
			to_cede_province_to_exarch = {
				CURRENT_OWNER = owner
				NEW_OWNER = event_target:to_exarch_province_recipient
			}
		}
	}
}

		to_exarch_effect_including_inactive = {
			EFFECT = "
				set_variable = {
					which = to_exarch_needed_provinces_for_merchant
					value = 0
				}
				set_variable = {
					which = to_exarch_owned_provinces_for_merchant
					value = 0
				}
				set_variable = {
					which = to_exarch_needed_provinces_for_merchant_2
					value = 0
				}
				set_variable = {
					which = to_exarch_owned_provinces_for_merchant_2
					value = 0
				}
				"
		}
		every_province = {
			limit = {
				is_empty = no
				is_colony = no
				OR = {
					continent = europe
					continent = africa
					continent = asia
				}
				NOT = { to_province_always_counts_for_exarch_merchant = yes }
			}
			to_exarch_merchant_effect_from_province = {
				EFFECT = "
				to_setup_exarch_merchant_counters_province = yes
				"
			}
			to_exarch_merchant_2_effect_from_province = {
				EFFECT = "
				to_setup_exarch_merchant_2_counters_province = yes
				"
			}
		}
	}
}

to_fix_exarch_merchant_counters = {
	if = {
		limit = {
			to_has_exarch_government_reform = yes
			OR = {
				is_variable_equal = {
					which = to_exarch_needed_provinces_for_merchant
					value = 0
				}
				AND = {
					to_exarch_has_merchant_land_2 = yes
					is_variable_equal = {
						which = to_exarch_needed_provinces_for_merchant_2
						value = 0
					}
				}
			}
		}
		to_setup_exarch_merchant_counter = yes
	}
}

to_setup_exarch_merchant_counter = {

		which = to_exarch_needed_provinces_for_merchant
		value = 0
	}
	set_variable = {

			NOT = { to_province_always_counts_for_exarch_merchant = yes }

				SCOPE = "PREV"
			}

	if = {
		limit = {
			owner = {
				is_subject_of_type = elysian_subject
			}
			NOT = { has_province_flag = to_was_counted_for_exarch_merchant }
		}
		if = {
			limit = {
				NOT = { to_province_always_counts_for_exarch_merchant = yes }
			}
			if = {
				limit = {
					to_exarch_merchant_land = {
						SCOPE = "owner"
					}
				}
				set_province_flag = to_was_counted_for_exarch_merchant
				owner = {
					change_variable = {
						which = to_exarch_owned_provinces_for_merchant
						value = 1
					}
				}
			}
			else_if = {
				limit = {
					owner = {
						has_country_flag = to_exarch_two_merchants
					}
					to_exarch_merchant_land_2 = {
						SCOPE = "owner"
					}
				}
				set_province_flag = to_was_counted_for_exarch_merchant
				owner = {
					change_variable = {
						which = to_exarch_owned_provinces_for_merchant_2
						value = 1
					}
				}
			}
		}
	}
	else_if = {
		limit = {
			FROM = {
				is_subject_of_type = elysian_subject
			}
			has_province_flag = to_was_counted_for_exarch_merchant
		}
		clr_province_flag = to_was_counted_for_exarch_merchant
		if = {
			limit = {
				FROM = { has_country_flag = to_exarch_two_merchants }
				to_exarch_merchant_land_2 = {
					SCOPE = "FROM"
				}
			}
			FROM = {
				change_variable = {
					which = to_exarch_owned_provinces_for_merchant_2
					value = -1
				}
			}
		}
		else = {
			FROM = {
				change_variable = {
					which = to_exarch_owned_provinces_for_merchant
					value = -1
				}
			}
		}
	}
}

to_release_exarch_from_province = {
	hidden_effect = {
		to_save_years_of_nationalism = yes
		if = {
			limit = {
				is_part_of_hre = yes
			}
			set_in_empire = no
		}
		clr_province_flag = nhs_exarch_province_exemption

		if = {
			limit = {
				owner = {
					PREV = {
						is_state_core = PREV
					}
				}
			}
			remove_core = $CURRENT_OWNER$
			if = {
				limit = {
					NOT = { is_state_core = $EXARCH_SCOPE$ }
				}
				add_core = $EXARCH_SCOPE$
			}
		}
		else_if = {
			limit = {
				owner = {
					PREV = {
						is_territorial_core = PREV
					}
				}
			}
			remove_core = $CURRENT_OWNER$
			set_province_flag = to_add_territorial_core
		}
		else_if = {
			limit = {
				owner = {
					PREV = {
						is_permanent_claim = PREV
					}
				}
			}
			remove_claim = $CURRENT_OWNER$
			set_province_flag = to_add_permanent_claim
		}
		else_if = {
			limit = {
				owner = {
					PREV = {
						is_claim = PREV
					}
				}
			}
			remove_claim = $CURRENT_OWNER$
			set_province_flag = to_add_claim
		}

		if = {
			limit = {
				NOT = { is_core = $EXARCH_SCOPE$ }
			}
			set_province_flag = to_temp_core
			add_core = $EXARCH_SCOPE$
		}
		
		$CURRENT_OWNER$ = {
			release = $EXARCH_SCOPE$
			create_subject = {
				subject_type = elysian_subject
				subject = $EXARCH_SCOPE$
			}
		}

		if = {
			limit = {
				has_province_flag = to_temp_core
			}
			clr_province_flag = to_temp_core
			remove_core = $EXARCH_SCOPE$
		}

		if = {
			limit = {
				has_province_flag = to_add_territorial_core
			}
			clr_province_flag = to_add_territorial_core
			add_territorial_core = $EXARCH_SCOPE$
		}
		else_if = {
			limit = {
				has_province_flag = to_add_permanent_claim
			}
			clr_province_flag = to_add_permanent_claim
			add_permanent_claim = $EXARCH_SCOPE$
		}
		else_if = {
			limit = {
				has_province_flag = to_add_claim
			}
			clr_province_flag = to_add_claim
			add_claim = $EXARCH_SCOPE$
		}
		set_province_flag = to_separatism_restore_pending
		province_event = { id = nhs2_ely_marches.5 days = 1 }
		to_load_years_of_nationalism = yes
		if = {
			limit = {
				$EXARCH_SCOPE$ = {
					NOT = { accepted_culture = PREV }
				}
			}
			$EXARCH_SCOPE$ = {
				add_accepted_culture = PREV
			}
		}
	}
}
