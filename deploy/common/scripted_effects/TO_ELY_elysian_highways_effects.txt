	clr_global_flag = to_elysian_highway_$HIGHWAY$_busy

to_elysian_highways_progress_routine_province = {
	# to_elysian_highway_all_effect
	if = {
		limit = {
			to_has_elysian_highways_potential = yes
		}
		[[HIGHWAY]
		if = {
			limit = {
				has_global_flag = to_elysian_highway_$HIGHWAY$_busy
			}
			set_province_flag = to_local_elysian_highway_busy
			set_province_flag = to_local_elysian_highway_busy_$HIGHWAY$
		}
		else = {
			clr_province_flag = to_local_elysian_highway_busy
			to_clear_highway_busy_flags = yes
		}
		]
		if = {
			limit = {
				to_elysian_highways_increasing = yes
			}
			owner = {
				save_event_target_as = to_elysian_highways_owner
			}
			[[HIGHWAY]
			set_global_flag = to_elysian_highway_$HIGHWAY$_busy
			save_global_event_target_as = to_elysian_highway_$HIGHWAY$_current
			]
			to_increase_elysian_highways_progress = { 
				OWNER_TARGET = to_elysian_highways_owner
				[[HIGHWAY]
				HIGHWAY = $HIGHWAY$
				]
			}
		}
		else = {
			to_elysian_highways_inactive_setup = yes
		}
	}
	else_if = {
		limit = {
			NOT = { to_has_elysian_highway_modifier = yes }
			NOT = { has_province_modifier = to_elysian_highway_progress_cd }
		}
		to_remove_elysian_highways_modifiers = yes
		clr_province_flag = to_elysian_highways_progressing
		remove_province_modifier = nhs_elysian_highways_inactive
	}
}

to_elysian_highways_inactive_setup = {
	if = {
		limit = {
			is_variable_equal = {
				which = to_elysian_highway_progress
				value = 0
			}
		}
		set_variable = {
			which = to_elysian_highway_progress
			value = 0
		}
	}
	clr_province_flag = to_elysian_highways_progressing
	to_remove_elysian_highways_modifiers = yes
	add_permanent_province_modifier = {
		name = nhs_elysian_highways_inactive
		duration = -1
	}
}

to_increase_elysian_highways_progress = {
	remove_province_modifier = nhs_elysian_highways_inactive
	set_province_flag = to_elysian_highways_progressing
	add_province_modifier = {
		name = to_elysian_highway_progress_cd
		duration = 5
		hidden = yes
	}
	set_variable = {
		which = to_elysian_highways_flat_increase
		value = 0
	}
	set_variable = {
		which = to_elysian_highways_modifier
		value = 1
	}
	set_variable = {
		which = to_elysian_highways_base
		value = 50
	}
	# Flat Increase
	change_variable = {
		which = to_elysian_highways_flat_increase
		which = to_elysian_highways_base
	}

	# Modifiers
	set_variable = {
		which = to_elysian_highways_development
		value = 0
	}
	export_to_variable = {
		variable_name = to_elysian_highways_development
		value = trigger_value:development
	}
	multiply_variable = {
		which = to_elysian_highways_development
		value = 0.01
	}
	change_variable = {
		which = to_elysian_highways_modifier
		which = to_elysian_highways_development
	}

	export_to_variable = {
		which = to_elysian_highways_build_time
		value = modifier:build_time
		who = event_target:$OWNER_TARGET$
	}
	export_to_variable = {
		which = to_elysian_highways_local_build_time
		value = modifier:local_build_time
	}
	change_variable = {
		which = to_elysian_highways_build_time
		which = to_elysian_highways_local_build_time
	}
	multiply_variable = {
		which = to_elysian_highways_build_time
		value = -1
	}
	change_variable = {
		which = to_elysian_highways_modifier
		which = to_elysian_highways_build_time
	}

	if = {
		limit = {
			NOT = {
				check_variable = {
					which = to_elysian_highways_modifier
					value = -0.90
				}
			}
		}
		set_variable = {
			which = to_elysian_highways_modifier
			value = -0.90
		}
	}
	# Total
	set_variable = {
		which = to_elysian_highways_increase_yearly
		which = to_elysian_highways_flat_increase
	}
	multiply_variable = {
		which = to_elysian_highways_increase_yearly
		which = to_elysian_highways_modifier
	}
	set_variable = {
		which = to_elysian_highways_increase_monthly
		which = to_elysian_highways_increase_yearly
	}
	divide_variable = {
		which = to_elysian_highways_increase_monthly
		value = 12
	}

	change_variable = {
		which = to_elysian_highway_progress
		which = to_elysian_highways_increase_monthly
	}
	if = {
		limit = {
			check_variable = {
				which = to_elysian_highway_progress
				value = 100
			}
		}
		to_complete_elysian_highway = {
			[[HIGHWAY]
			HIGHWAY = $HIGHWAY$
			]
		}
	}
	else = {
		set_variable = {
			which = to_elysian_highways_remaining_months
			value = 100
		}
		subtract_variable = {
			which = to_elysian_highways_remaining_months
			which = to_elysian_highway_progress
		}
		divide_variable = {
			which = to_elysian_highways_remaining_months
			which = to_elysian_highways_increase_monthly
		}
		to_round_up_variable = { VARIABLE = to_elysian_highways_remaining_months }

		set_variable = {
			which = to_elysian_highways_remaining_years
			value = 100
		}
		subtract_variable = {
			which = to_elysian_highways_remaining_years
			which = to_elysian_highway_progress
		}
		divide_variable = {
			which = to_elysian_highways_remaining_years
			which = to_elysian_highways_increase_yearly
		}
		to_round_down_variable_one_decimal = { VARIABLE = to_elysian_highways_increase_yearly }
		to_round_up_variable = { VARIABLE = to_elysian_highways_remaining_years }

		change_variable = {
			which = to_elysian_highways_modifier
			value = -1
		}
		multiply_variable = {
			which = to_elysian_highways_modifier
			value = 100
		}
		multiply_variable = {
			which = to_elysian_highways_build_time
			value = 100
		}
		multiply_variable = {
			which = to_elysian_highways_development
			value = 100
		}
		to_set_correct_elysian_highways_modifier = { MODIFIER_BASE_NAME = nhs_elysian_highways_inprogress }

		set_variable = {
			which = to_elysian_highway_progress_display
			which = to_elysian_highway_progress
		}
		to_round_down_variable_one_decimal = { VARIABLE = to_elysian_highway_progress_display }
	}
}

to_complete_elysian_highway = {
	to_remove_elysian_highway_variables = yes
	remove_province_modifier = nhs_elysian_highways_inactive
	[[HIGHWAY]
	if = {
		limit = {
			has_global_flag = to_elysian_highway_$HIGHWAY$_busy
		}
		clr_global_flag = to_elysian_highway_$HIGHWAY$_busy
		if = {
			limit = {
				event_target:to_elysian_highway_$HIGHWAY$_current = {
					province_id = PREV
				}
			}
			clear_global_event_target = to_elysian_highway_$HIGHWAY$_current
		}
	}	
	]
	if = {
		limit = {
			has_global_flag = to_elysian_highways_project_tier_3
		}
		add_permanent_province_modifier = {
			name = nhs_elysian_highways_3
			duration = -1
		}
	}
	else_if = {
		limit = {
			has_global_flag = to_elysian_highways_project_tier_2
		}
		add_permanent_province_modifier = {
			name = nhs_elysian_highways_2
			duration = -1
		}
	}
	else = {
		add_permanent_province_modifier = {
			name = nhs_elysian_highways
			duration = -1
		}
	}
}
