to_eligible_for_cultureshock = {
	is_city = yes
	OR = {
		culture = greek
		culture = pontic
		culture = gothic
	}

	owner = {
		nhs_is_elysian_tag_or_subject = yes
	}

	NOT = {
		to_has_cultureshock = yes
	}
}

to_has_cultureshock = {
	OR = {
		has_province_modifier = nhs_cultureshock_vbad
		has_province_modifier = nhs_cultureshock_bad
		has_province_modifier = nhs_cultureshock_medium
		has_province_modifier = nhs_cultureshock_light
		has_province_modifier = nhs_cultureshock_vlight
	}
}
