	if = {
		limit = {
			has_country_flag = nhs_disable_noble_privileges
		}
		custom_trigger_tooltip = {
			tooltip = to_nhs_nobles_had_event_check_tt
			always = no
		}
	}
}

to_estate_will_revolt = {
	has_estate = $ESTATE$
	OR = {
		NOT = {
			estate_loyalty = {
				estate = $ESTATE$
				loyalty = 30
			}
		}
		AND = {
			NOT = { estate_has_exempt_from_seize_land_privilege = { estate = $ESTATE$ } }
			NOT = {
				estate_loyalty = {
					estate = $ESTATE$
					loyalty = 50
				}
			}
		}
	}
}
