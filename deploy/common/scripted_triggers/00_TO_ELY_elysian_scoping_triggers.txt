
# Check if CONDITION matches any Elysian nation
# Use: nhs_check_all_elysian_tags = { CONDITION = tag }
nhs_check_all_elysian_tags = {
	OR = {
		$CONDITION$ = ELY  # Elysian Empire
		$CONDITION$ = NHS  # Elysia
		$CONDITION$ = ELR  # Elysian Noble Republic
		$CONDITION$ = ERR  # Revolutionary Elysia
		$CONDITION$ = ERF  # First Elysian Republic
	}
}

# Check if CONDITION matches any Elysian nation, including Creta Minor and culture primaries
# Use: nhs_check_all_elysian_tags_expanded = { CONDITION = tag }
nhs_check_all_elysian_tags_expanded = {
	OR = {
		nhs_check_all_elysian_tags = { CONDITION = $CONDITION$ }
		$CONDITION$ = CM1  # Creta Minor
		# $CONDITION$ = ELF  # Elysian settlers
		$CONDITION$ = ELA  # Elysian Federation
		$CONDITION$ = KYK  # Kykladia
		$CONDITION$ = SKY  # Scythia
		$CONDITION$ = BOR  # Borealia
		$CONDITION$ = HBR  # Hyperborealia
		$CONDITION$ = HSP  # Hesperidia
		$CONDITION$ = LA1  # Lakonia
		$CONDITION$ = VAE  # Varangia
		$CONDITION$ = HBN  # Hybernica
		AND = {
			OR = {
				nhs_post_fragmentation_culturally_elysian = yes
				AND = {
					primary_culture = greek
					NOT = { has_global_flag = nhs_romaic_culture_change }
					technology_group = exiled_greek
				}
			}
			OR = {
				is_colonial_nation = yes
				is_former_colonial_nation = yes
			}
		}
	}
}

nhs_check_all_elysian_tags_with_if = {
	if = {
		limit = {
			exists = ELY
		}
		$CONDITION$ = ELY
	}
	else_if = {
		limit = {
			exists = NHS
		}
		$CONDITION$ = NHS
	}
	else_if = {
		limit = {
			exists = ELR
		}
		$CONDITION$ = ELR
	}
	else_if = {
		limit = {
			exists = ERR
		}
		$CONDITION$ = ERR
	}
	else_if = {
		limit = {
			exists = ERF
		}
		$CONDITION$ = ERF
	}
	else_if = {
		limit = {
			has_global_flag = nhs_romaic_to_elysian
		}
		$CONDITION$ = ELY
	}
	else = {
		$CONDITION$ = NHS
	}
}

nhs_msg_claim_elysian_throne_trigger_ely = {
	OR = {
		AND = {
			exists = ELY
			OR = {
				ELY = {
					AND = {
						has_regency = yes
						has_consort_regency = no
					}
				}
				AND = {
					ai = no
					total_development = 2000
					legitimacy = 100
				}
				151 = { #implies Elysia got Constantinople, but lost it again
					is_core = ELY
					NOT = { owned_by = ELY }
				}
				953 = { NOT = { owned_by = ELY } }
			}
		}
		AND = {
			exists = NHS
			OR = {
				NHS = {
					AND = {
						has_regency = yes
						has_consort_regency = no
					}
				}
				AND = {
					ai = no
					total_development = 2000
					legitimacy = 100
				}
				151 = { #implies Elysia got Constantinople, but lost it again
					is_core = NHS
					NOT = { owned_by = NHS }
				}
				953 = { NOT = { owned_by = NHS } }
			}
		}
		AND = {
			OR = {
				exists = ELR
				exists = ERR
				exists = ERF
			}
			NOT = { exists = ELY }
			NOT = { exists = NHS }
		}
	}
}
