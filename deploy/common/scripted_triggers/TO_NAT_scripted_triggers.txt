to_will_consider_joining_confederation_development = {
	variable_arithmetic_trigger = {
		export_to_variable = {
			variable_name = to_total_dev_with_subjects_$TARGET$
			value = trigger_value:total_own_and_non_tributary_subject_development
			who = $TARGET$
		}
		export_to_variable = {
			variable_name = to_total_dev_$LEADER$
			value = trigger_value:total_development
			who = $LEADER$
		}
		export_to_variable = {
			variable_name = to_subject_dev_$LEADER$
			value = trigger_value:total_own_and_non_tributary_subject_development
			who = $LEADER$
		}
		subtract_variable = {
			which = to_subject_dev_$LEADER$
			which = to_total_dev_$LEADER$
		}
		divide_variable = {
			which = to_subject_dev_$LEADER$
			value = 4
		}
		change_variable = {
			which = to_total_dev_$LEADER$
			which = to_subject_dev_$LEADER$
		}
		multiply_variable = {
			which = to_total_dev_$LEADER$
			value = 1.25
		}
		check_variable = {
			which = to_total_dev_$LEADER$
			which = to_total_dev_with_subjects_$TARGET$
		}
	}
}
