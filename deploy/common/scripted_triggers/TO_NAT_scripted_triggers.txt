	OR = {
		government = native
		has_reform = native_basic_reform
		has_reform = native_reformed_basic_reform
	}
}

	}

to_nat_is_major_native_nation = {
	OR = {
		total_development = 50
		custom_trigger_tooltip = {
			tooltip = to_nat_has_most_development_in_culture_tt
			NOT = {
				any_country = {
					NOT = { tag = ROOT }
					is_subject = no
					primary_culture = ROOT
					total_development = ROOT
				}
			}
		}
	}
}

	OR = {
		continent = north_america

		is_empty = yes

			is_native_government = yes

		}
	}
}

to_nat_province_displacement_origin_check = {
	is_colony = no
	NOT = { has_province_flag = nhs_is_colony }
	has_province_modifier = to_nat_tribal_development
	has_owner_culture = no
}

to_nat_province_displacement_target_check = {
	is_colony = no
	OR = {
		NOT = { development = 12 }
		AND = {
			is_empty = no
			owner = {
				to_nat_has_any_tier_8_native_government_reform = yes
			}
		}
	}
	$PROVINCE_SCOPE$ = {
		NOT = { province_id = PREV }
	}
	[[ALTERNATIVE_TARGET_1]
		event_target:$ALTERNATIVE_TARGET_1$ = {
			NOT = { province_id = PREV }
		}
	]
	[[ALTERNATIVE_TARGET_2]
		event_target:$ALTERNATIVE_TARGET_2$ = {
			NOT = { province_id = PREV }
		}
	]
}
