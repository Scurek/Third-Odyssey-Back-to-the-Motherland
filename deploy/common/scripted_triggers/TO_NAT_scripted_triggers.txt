is_native_government = {
	OR = {
		government = native
		has_reform = native_basic_reform
		has_reform = native_reformed_basic_reform
	}
}

to_nat_has_any_tier_4_native_government_reform = {
	OR = {
		has_reform = to_native_reform_consensus_government
		has_reform = to_native_reform_law_of_peace
		has_reform = to_native_reform_grand_chiefdom
		has_reform = to_native_reform_kingdom
	}
}

to_nat_estates_enabled = {
	OR = {
		has_reform = native_reformed_basic_reform
		AND = {
			has_reform = native_basic_reform
			to_nat_has_any_tier_4_native_government_reform = yes
		}
	}
}

to_nat_can_have_estate = {
	has_country_flag = to_nat_estate_enabled_$ESTATE$
}

to_nat_is_major_native_nation = {
	OR = {
		total_development = 50
		custom_trigger_tooltip = {
			tooltip = to_nat_has_most_development_in_culture_tt
			NOT = {
				any_country = {
					NOT = { tag = ROOT }
					is_subject = no
					primary_culture = ROOT
					total_development = ROOT
				}
			}
		}
	}
}

	OR = {
		AND = {
			is_colony = yes

		}

	}
