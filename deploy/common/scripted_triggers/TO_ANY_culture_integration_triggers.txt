nhs_mesoamerican_integration_province = {
	mesoamerican_native_culture = yes
	NOT = { culture_group = spartakian_group }
	NOT = { has_province_modifier = to_nat_tribal_development }
	NOT = { has_province_modifier = to_nat_had_tribal_development }
}

nhs_spartan_integration_province = {
	culture_group = spartakian_group
}

nhs_norse_integration_province = {
	culture_group = norse_g
}

to_has_culture_assimilation_potential = {
	is_empty = no
	is_city = yes
	OR = {
		has_province_modifier = to_nat_tribal_development
		AND = {
			owner = {
				has_estate = estate_all_natives
			}
			nhs_culturally_native = yes
			has_province_modifier = to_nat_had_tribal_development
		}
		AND = {
			owner = {
				has_country_flag = nhs_mesoamerican_integration
			}
			nhs_mesoamerican_integration_province = yes
		}
		AND = {
			owner = {
				has_country_flag = nhs_mesoamerican_integration
			}
			nhs_spartan_integration_province = yes
		}
		AND = {
			owner = {
				has_country_flag = nhs_mesoamerican_integration
			}
			nhs_norse_integration_province = yes
		}
	}
	OR = {
		NOT = { culture = elysameric }
		NOT = { owner = { culture_group = elysian_group } }
	}
	OR = {
		NOT = { culture = skraeling }
		NOT = { owner = { culture_group = norse_g } }
	}
	owner = {
		NOT = { has_native_government = yes }
	}
	NOT = { has_province_flag = nhs_got_core_culture_change_delay }
}

to_culture_assimilation_increasing = {
	owner = {
		is_core = PREV
	}
	OR = {
		has_owner_religion = yes
		tolerance_to_this = 0
	}
	NOT = { nationalism = 1 }
}

to_has_tribal_integration = {
	has_province_modifier = to_nat_tribal_development
	OR = {
		NOT = { nhs_culturally_native = yes }
		owner = {
			culture_group = PREV
		}
	}
}
