			NOT = { has_country_modifier = nhs_hrv_peaceful_resolution }
			NOT = { has_country_modifier = nhs_hrv_tolerance }
		}	
	}
	else = {
		if = {
			limit = {
				has_country_modifier = nhs_hcd_tolerant_approach
			}
			NOT = { has_country_modifier = nhs_hcd_tolerant_approach }
		}
		if = {
			limit = {
				has_country_modifier = nhs_hrv_peaceful_resolution
			}
			NOT = { has_country_modifier = nhs_hrv_peaceful_resolution }
		}
		if = {
			limit = {
				has_country_modifier = nhs_hrv_tolerance
			}
			NOT = { has_country_modifier = nhs_hrv_tolerance }
		}
	}
}
