custom_button = {
    name = to_trade_expeditions_button_send
    tooltip = to_trade_expeditions_button_send_tt
    potential = {
        FROM = {
            to_unlocked_trade_expeditions = yes
        }
    }
    trigger = {
        custom_trigger_tooltip = {
            tooltip = to_is_not_in_chesapeake_bay_tn_tt
            NOT = { same_trade_node_as = 953 }
        }
        is_not_inland_trade_node_province = yes
        trade_range = FROM
        custom_trigger_tooltip = {
            tooltip = to_no_trade_expedition_tt
            FROM = {
                OR = {
                    NOT = { has_country_flag = to_send_trade_expedition }
                    had_country_flag = {
                        flag = to_send_trade_expedition
                        days = 7300
                    }
                }
            }
        }
        if = {
            limit = {
                has_country_flag = to_has_trade_expedition_finished_event
            }
            custom_trigger_tooltip = {
                tooltip = to_expedition_destination_event_not_active_tt
                always = no
            }
        }
    }
    effect = {
        FROM = {
            clr_country_flag = to_send_trade_expedition
            set_country_flag = to_send_trade_expedition
        }
        save_global_event_target_as = to_trade_expedition_target
        add_trade_modifier = {
            who = FROM
            duration = 7300
            power = 50
            key = TO_TRADE_EXPEDITION_TRADE_MOD
        }
        if = {
            limit = {
                any_country = {
                    home_trade_node = {
                        same_trade_node_as = ROOT
                    }
                    NOT = { tag = FROM }
                    FROM = {
                        knows_country = PREV
                    }
                }
            }
            custom_tooltip = nhs_new_line_tt
            every_country = {
                limit = {
                    home_trade_node = {
                        same_trade_node_as = ROOT
                    }
                    NOT = { tag = FROM }
                    FROM = {
                        knows_country = PREV
                    }
                }
                custom_tooltip = to_add_trade_expedition_opinion_of_tt
                hidden_effect = {
                    if = {
                        limit = {
                            NOT = { trade_embargoing = FROM }
                        }
                        add_opinion = {
                            who = FROM
                            modifier = to_trade_expedition_opinion
                        }
                    }
                }
            }
            custom_tooltip = nhs_new_line_tt
        }
        custom_tooltip = to_add_trade_expedition_modifiers_tt
        hidden_effect = {
            every_province = {
                limit = {
                    same_trade_node_as = ROOT
                    NOT = { country_or_non_sovereign_subject_holds = FROM }
                    owner = {
                        NOT = { trade_embargoing = FROM }
                    }
                }
                add_permanent_province_modifier = {
                    name = to_trade_expedition_province_modifier
                    duration = 7300
                }
            }

            FROM = {
                country_event = { id = nhs_monuments.11 }
                country_event = {
                    id = nhs_monuments.10
                    days = 7300
                }
            }
        }
    }
}

    name = to_trade_expeditions_button_cancel
