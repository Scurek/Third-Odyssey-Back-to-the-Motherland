nhs_gibraltar_mod = {
	potential = {
	}
	
	#viewer = {
	#}
	trigger = {
		custom_trigger_tooltip = {
			tooltip = nhs_owned_by_elysia_or_subject_tt
			owner = {
				nhs_is_elysian_tag_or_subject = yes
			}
		}
	}
	province_trade_power_value = 10
	local_sailors = 500
	local_ship_repair = 0.5
	ship_recruit_speed = -0.25
	fort_level = 1
	local_defensiveness = 0.5
	local_hostile_attrition = 2
	#Province scope
	on_activation = {
		owner = {
			if = {
				limit = {
					OR = {
						AND = {
							is_subject = yes
							NOT = {
								overlord = {
									226 = {
										has_trade_modifier = {
											who = PREV
											key = CONTROLS_THE_ROCK
										}
									}
								}
							}
							
						}
						AND = {
							is_subject = no
							NOT = {
								226 = {
									has_trade_modifier = {
										who = PREV
										key = CONTROLS_THE_ROCK
									}
								}
							}
						}
					}
				}
				226 = {
					add_trade_modifier = {
						who = ROOT
						duration = -1
						power = 30
						key = CONTROLS_THE_ROCK
					}
				}
			}
		}
	}
	on_deactivation = {
		owner = {
			random_country = {
				limit = {
					NOT = { overlord_of = PREV }
					NOT = { tag = PREV }
					226 = {
						has_trade_modifier = {
							who = PREV
							key = CONTROLS_THE_ROCK
						}
					}
				}
				226 = {
					remove_trade_modifier = {
						who = PREV
						key = CONTROLS_THE_ROCK
					}
				}
			}
		}
	}
}

nhs2_lantent_coal_tm = {
	potential = {
		if = {
			limit = {
				NOT = { trade_goods = coal_2 }
			}
			OR = {
				development = 20
				owner = {
					full_idea_group = innovativeness_ideas
				}
			}
			provincial_institution_progress = {
				which = enlightenment
				value = 100
			}
			owner = { has_institution = enlightenment }
		}
		else = {
			NOT = {
				AND = {
					OR = {
						development = 20
						owner = {
							full_idea_group = innovativeness_ideas
						}
					}
					provincial_institution_progress = {
						which = enlightenment
						value = 100
					}
					owner = { has_institution = enlightenment }
				}
			}
		}
	}
	trigger = {
		always = yes
	}
	
	on_activation = {
		hidden_effect = {
			if = {
				limit = {
					NOT = { trade_goods = coal_2 }
				}
				nhs_save_previous_goods = yes
				remove_province_modifier = nhs2_lantent_coal
				change_trade_goods = coal_2
			}
			else = {
				nhs_load_previous_goods = yes
				add_permanent_province_modifier = {
					name = nhs2_lantent_coal
					duration = -1
				}
			}
			
		}
	}
