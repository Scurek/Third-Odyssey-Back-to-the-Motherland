# Native estate for exiled nations (Elysia, Vinland, Spartakon, but also Creta Minor, Shqiperi, Varangian Order...)
# The estate morphs over time in base loyalty, influence and privileges.

estate_all_natives = {
    icon = 12
    color = { 150 95 50 }

	# Used for when declaring independence
	independence_government = republic
	starting_reform = native_council_after
	independence_tech_group = hellenised_american

	province_independence_weight = {
		factor = 1
		modifier = {
			factor = 2
			any_neighbor_province = {
				native_culture = yes
			}
			native_culture = yes
		}
		modifier = {
			factor = 2
			native_culture = yes
		}
	}

	# If true, country will get estate
	# Basically applies to all exiles regardless of province cultures; represents all natives in populace.
    trigger = {
		OR = {
        	has_country_flag = native_council_enabled_early
        	has_country_flag = native_council_enabled_mid
        	has_country_flag = native_council_enabled_late
		}
    }

	# These scale with loyalty & power
	country_modifier_happy = {
	    # None!
	}
	country_modifier_neutral = {
	    diplomatic_reputation = -1
		manpower_recovery_speed = -0.25
	}
	country_modifier_angry = {
		diplomatic_reputation = -2
		manpower_recovery_speed = -0.5
	}

	# Influence modifiers

	base_influence = 30
	influence_from_dev_modifier = 1.0

	influence_modifier = {
		desc = EST_VAL_ESTATE_NATIVES_EARLY_INFLUENCE
		trigger = {
			has_country_flag = native_council_enabled_early
		}	
		influence = 20
	}
	
	influence_modifier = {
		desc = EST_VAL_ESTATE_NATIVES_MID_INFLUENCE
		trigger = {
			has_country_flag = native_council_enabled_mid
		}	
		influence = 10
	}

	# Loyalty modifiers
	
	loyalty_modifier = {
		desc = EST_VAL_ESTATE_NATIVES_EARLY_LOYALTY
		trigger = {
			has_country_flag = native_council_enabled_early
		}
		loyalty = -20
	}
	
	loyalty_modifier = {
		desc = EST_VAL_ESTATE_NATIVES_MID_LOYALTY
		trigger = {
			has_country_flag = native_council_enabled_mid
		}
		loyalty = -10
	}


	# Custom names

	custom_name = {
		desc = estate_natives_elysian
		trigger = {
			nhs_culturally_elysian  = yes
		}
	}
	custom_name = {
		desc = estate_natives_norse
		trigger = {
			nhs_culturally_norse = yes
		}
	}
	custom_name = {
		desc = estate_natives_spartan
		trigger = {
			nhs_culturally_spartan = yes
		}
	}
	custom_name = {
		desc = estate_natives_shqiperi
		trigger = {
			nhs_culturally_albanian = yes
		}
	}

	privileges = {
		# -- Static --

		# Generic
		nhs_native_priv_local_councils_empowered


		# -- Early --
		
		# Generic
		nhs_native_priv_expansionist_coexistance_early
		nhs_native_priv_increase_tribute_early
		# Elysia
		nhs_native_priv_minor_pax_romana_early
		nhs_native_priv_major_pax_romana_early
		nhs_native_priv_encourage_barbaroi_officials_early
		nhs_native_priv_divide_et_impera_early

		# -- Mid --

		# Generic
		nhs_native_priv_expansionist_coexistance_mid
		nhs_native_priv_increase_tribute_mid
		# Elysia
		nhs_native_priv_major_pax_romana_mid
		nhs_native_priv_symmachoi_noble_offices_mid
		nhs_native_priv_designate_development_zones_mid
		nhs_native_priv_divide_et_impera_mid

		# -- Late --

		# Generic
		nhs_native_priv_expansionist_coexistance_late
		nhs_native_priv_increase_tribute_late
		# Elysia
		nhs_native_priv_designate_development_zones_late
	}

}
