# Native estate for exiled nations (Elysia, Vinland, Spartakon, but also Create Minor, Shqiperi, Varangian Order...)
# The various estates are basically duplicates. Only happy/neutral/angry modifiers are different, and some privileges.

estate_all_natives_early = {
    icon = 12
    color = { 150 95 50 }

	# Used for when declaring independence
	independence_government = republic
	starting_reform = native_council_after
	independence_tech_group = hellenised_american

	province_independence_weight = {
		factor = 1
		modifier = {
			factor = 2
			any_neighbor_province = {
				native_culture = yes
			}
			native_culture = yes
		}
		modifier = {
			factor = 2
			native_culture = yes
		}
	}

	# If true, country will get estate
	# Basically applies to all exiles regardless of province cultures; represents all natives in populace.
    trigger = {
        has_country_flag = native_council_enabled_early
        nhs_check_all_exiles = { CONDITION = tag }
        capital_scope = {
            OR = {
                continent = north_america
                continent = south_america
            }
        }
    }

	# These scale with loyalty & power
	country_modifier_happy = {
	    # None!
	}
	country_modifier_neutral = {
	    diplomatic_reputation = -1
		manpower_recovery_speed = -0.25
		# "regional aggression" + 50%
	}
	country_modifier_angry = {
		diplomatic_reputation = -2
		manpower_recovery_speed = -0.5
		# "regional aggression" + 100%
	}


	custom_name = {
		desc = estate_natives_elysian
		trigger = {
			nhs_culturally_elysian  = yes
		}
	}
	custom_name = {
		desc = estate_natives_norse
		trigger = {
			nhs_culturally_norse = yes
		}
	}
	custom_name = {
		desc = estate_natives_spartan
		trigger = {
			nhs_culturally_spartan = yes
		}
	}
	custom_name = {
		desc = estate_natives_shqiperi
		trigger = {
			nhs_culturally_albanian = yes
		}
	}

	privileges = {
		nhs_native_priv_expansionist_coexistance_early
		bns_native_priv_local_councils_recognized_early
		bns_native_priv_minor_pax_romana_early
		bns_native_priv_major_pax_romana_early
		bns_native_priv_increase_tribute_early
		bns_native_priv_encourage_barbaroi_officials_early
		bns_native_priv_divide_et_impera_early
	}

	base_influence = 20

}
