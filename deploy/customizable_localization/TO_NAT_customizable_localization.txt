defined_text = {
	name = GetTrueFalseOrEstate
	random = no
	
	text = {
		localisation_key = to_nat_y_icon_tt
		trigger = {
			owner = {
				OR = {
					AND = {
						has_estate = estate_all_natives
						# NOT = { has_estate_privilege = nhs_native_priv_ancestral_land }
						NOT = {
							estate_loyalty = {
								estate = estate_all_natives
								loyalty = 30
							}
						}
					}
					AND = {
						NOT = { has_estate = estate_all_natives }
						NOT = {
							has_global_modifier_value = {
								which = native_assimilation
								value = 1.0
							}
						}
					}
				}
			}
		}
	}
	
	text = {
		localisation_key = to_nat_x_icon_tt
	}
}
