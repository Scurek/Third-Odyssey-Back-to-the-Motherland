nhs_trigger_event = {
	if = {
		limit = {
			exists = NHS
		}
		NHS = {
			country_event = { id = $EVENT_ID$ }
		}
	}
	else_if = {
		limit = {
			exists = ELY
		}
		ELY = {
			country_event = { id = $EVENT_ID$ }
		}
	}
	else_if = {
		limit = {
			exists = ELR
		}
		ELR = {
			country_event = { id = $EVENT_ID$ }
		}
	}
	else_if = {
		limit = {
			exists = ERR
		}
		ERR = {
			country_event = { id = $EVENT_ID$ }
		}
	}
	else_if = {
		limit = {
			exists = ERF
		}
		ERF = {
			country_event = { id = $EVENT_ID$ }
		}
	}
}

nhs_elysian_nationalist_on_siege_rebel_effect = {

		}
		if = {
			limit = {
				any_province = {
					is_core = ELR
				}
			}
			add_core = ELR
		}
		else_if = {
			limit = {
				any_province = {
					is_core = ERR
				}
			}
			add_core = ERR
		}
		else_if = {
			limit = {
				any_province = {
					is_core = ERF
				}
			}
			add_core = ERF
		}
		else = {
			add_core = ELY
		}
	}
	else = {
		add_core = REB
	}
