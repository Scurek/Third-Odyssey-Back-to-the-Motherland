# AI colonisation help Vinland and Norse Colonies
province_event = {
	id = nhs_vov_ai_help.1
	title = nhs_vov_ai_help.1.t
	desc = nhs_vov_ai_help.1.d
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		owner = {
			OR = {
				tag = MK1
				tag = HLL
				tag = VIN
			}
		}
		has_empty_adjacent_province = yes
		owner = { NOT = { num_of_colonies = 1 } }
		owner = { ai = yes }
		owner = { NOT = { has_country_modifier = nhs2_ai_colonising_timer }	}
		owner = { num_of_colonists = 1 }
		NOT = { num_of_cities = 20 }
	}

	immediate = {
		owner = {
			add_country_modifier = {
				name = nhs2_ai_colonising_timer
				duration = 1
				hidden = yes
			}
		}
	}
	
	
	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			any_empty_neighbor_province = {
				province_id = 994
			}
		}
		random_empty_neighbor_province = {
			limit = {
				province_id = 994
			}
			create_colony = 50
		}
	}
	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			NOT = {
				any_empty_neighbor_province = {
					province_id = 994
				}
			}
			any_empty_neighbor_province = {
				province_group = thorfinn_bay
			}
		}
		random_empty_neighbor_province = {
			limit = {
				province_group = thorfinn_bay
			}
			create_colony = 50
		}
	}
	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			NOT = {
				any_empty_neighbor_province = {
					province_id = 994
				}
			}
			NOT = {
				any_empty_neighbor_province = {
					province_group = thorfinn_bay
				}
			}
			any_empty_neighbor_province = {
				region = canada_region
			}
		}
		random_empty_neighbor_province = {
			limit = {
				region = canada_region
			}
			create_colony = 50
		}
	}
	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			NOT = {
				any_empty_neighbor_province = {
					province_id = 994
				}
			}
			NOT = {
				any_empty_neighbor_province = {
					region = canada_region
				}
			}
			NOT = {
				any_empty_neighbor_province = {
					province_group = thorfinn_bay
				}
			}
			any_province = {
				area = greenland_area
				is_empty = yes
			}
			owner = { NOT = { has_country_modifier = nhs_vov_population } }
		}
		random_province = {
			limit = {
				area = greenland_area
			}
			create_colony = 10
		}
	}
	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			NOT = {
				any_empty_neighbor_province = {
					province_id = 994
				}
			}
			NOT = {
				any_empty_neighbor_province = {
					region = canada_region
				}
				any_province = {
					area = greenland_area
					is_empty = yes
				}
			}
		}
		random_empty_neighbor_province = {
			limit = {
				continent = north_america
			}
			create_colony = 10
		}
	}
	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			NOT = {
				any_empty_neighbor_province = {
					province_id = 994
				}
			}
			NOT = {
				any_empty_neighbor_province = {
					region = canada_region
				}
				any_province = {
					area = greenland_area
					is_empty = yes
				}
				any_empty_neighbor_province = {
					continent = north_america
				}
			}
		}
		random_empty_neighbor_province = {
			create_colony = 10
		}
	}
}


# AI colonisation help: Varangians
province_event = {
	id = nhs_vov_ai_help.2
	title = nhs_vov_ai_help.1.t
	desc = nhs_vov_ai_help.1.d
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		owner = {
			OR = {
				tag = VGD
			}
			NOT = { num_of_cities = 20 }
		}
		NOT = {
			any_empty_neighbor_province = {
				region = canada_region
			}
		}
		has_empty_adjacent_province = yes
		owner = { NOT = { num_of_colonies = 1 } }
		owner = { ai = yes }
		owner = { NOT = { has_country_modifier = nhs2_ai_colonising_timer }	}
		owner = { num_of_colonists = 1 }
	}

	immediate = {
		owner = {
			add_country_modifier = {
				name = nhs2_ai_colonising_timer
				duration = 1
				hidden = yes
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			any_empty_neighbor_province = {
				region = canada_region
				has_port = yes
			}
		}
		random_empty_neighbor_province = {
			limit = {
				region = canada_region
				has_port = yes
			}
			create_colony = 10
		}
	}
	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			NOT = {
				any_empty_neighbor_province = {
					region = canada_region
					has_port = yes
				}
			}
		}
		random_empty_neighbor_province = {
			limit = {
				region = canada_region
			}
			create_colony = 10
		}
	}
	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			NOT = {
				any_empty_neighbor_province = {
					region = canada_region
				}
			}
			any_empty_neighbor_province = {
				region = canada_region
				has_port = yes
			}
		}
		random_empty_neighbor_province = {
			limit = {
				region = canada_region
				has_port = yes
			}
			create_colony = 10
		}
	}
	option = {
		name = "nhs_vov_ai_help.1.a"
		trigger = {
			NOT = {
				any_empty_neighbor_province = {
					region = canada_region
				}
				any_empty_neighbor_province = {
					region = canada_region 
					has_port = yes
				}
			}
		}
		random_empty_neighbor_province = {
			limit = {
				region = canada_region
			}
		}
		create_colony = 10
	}
}

	id = nhs_vov_ai_help.4
	title = nhs_vov_ai_help.4.t
	desc = nhs_vov_ai_help.1.d
	picture = COLONIZATION_eventPicture
	
	hidden = yes
	
	trigger = {
		tag = MK1
		exists = MK1
		exists = HLL
		NOT = {
			has_global_flag = nhs_vov_war_brewing
			has_global_flag = nhs_vov_war_over
			has_country_modifier = nhs_truce_timer
		}
	}
	
	immediate = {
		hidden_effect = {
			add_truce_with = HLL
			add_country_modifier = {
				name = nhs_truce_timer
				duration = 1460
				hidden = yes
			}
		}
	}
	
	option = {
		name = "nhs_vov_ai_help.1.a"
	}
}
